72 template <typename T>
73 constexpr auto incomplete_gamma_inv_initial_val_2(T const a, T const p, T const tVal) noexcept -> T
74 { // a <= 1.0
75  return (
76  p < tVal ? // if
77  pow(p / tVal, T(1) / a)
78  :
79  // else
80  T(1) - log(T(1) - (p - tVal) / (T(1) - tVal))
81  );
82 }

195 template <typename T>
196 constexpr auto incomplete_gamma_inv_check(T const a, T const p) noexcept -> T
197 {
198  return ( // NaN check
199  any_nan(a, p) ? etl::numeric_limits<T>::quiet_NaN() :
200  //
201  etl::numeric_limits<T>::epsilon() > p ? T(0)
202  : p > T(1) ? etl::numeric_limits<T>::quiet_NaN()
203  : etl::numeric_limits<T>::epsilon() > abs(T(1) - p) ? etl::numeric_limits<T>::infinity()
204  :
