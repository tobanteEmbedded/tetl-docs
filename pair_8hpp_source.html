185 /// \brief Compares lhs and rhs lexicographically by operator<, that is,
186 /// compares the first elements and only if they are equivalent, compares the
187 /// second elements.
188 template <typename T1, typename T2>
189 constexpr auto operator<(pair<T1, T2> const& lhs, pair<T1, T2> const& rhs) -> bool
190 {
191  if (lhs.first < rhs.first) {
192  return true;

194  if (rhs.first < lhs.first) {
