35//
36// coefficients; see eq. 18.5.17b
37
38template <typename T>
39constexpr auto incomplete_beta_coef_even(T const a, T const b, T const z, int const k) noexcept -> T
40{
41 return (-z * (a + k) * (a + b + k) / ((a + 2 * k) * (a + 2 * k + T(1))));
42}
43
44template <typename T>
45constexpr auto incomplete_beta_coef_odd(T const a, T const b, T const z, int const k) noexcept -> T
46{
47 return (z * k * (b - k) / ((a + 2 * k - T(1)) * (a + 2 * k)));
48}
49
50template <typename T>
51constexpr auto incomplete_beta_coef(T const a, T const b, T const z, int const depth) noexcept -> T
52{
53 return (
54 !is_odd(depth) ? incomplete_beta_coef_even(a, b, z, depth / 2)
55 : incomplete_beta_coef_odd(a, b, z, (depth + 1) / 2)
56 );
57}
58
59//
60// update formulae for the modified Lentz method
61
62template <typename T>
63constexpr auto incomplete_beta_c_update(T const a, T const b, T const z, T const cJ, int const depth) noexcept -> T
64{
65 return (T(1) + incomplete_beta_coef(a, b, z, depth) / cJ);
66}
67
68template <typename T>
69constexpr auto incomplete_beta_d_update(T const a, T const b, T const z, T const dJ, int const depth) noexcept -> T
70{
71 return (T(1) / (T(1) + incomplete_beta_coef(a, b, z, depth) * dJ));
72}

77template <typename T>
78constexpr auto
79incomplete_beta_decision(T const a, T const b, T const z, T const cJ, T const dJ, T const fJ, int const depth) noexcept
80 -> T
81{
82 return ( // tolerance check
83 abs(cJ * dJ - T(1)) < GCEM_INCML_BETA_TOL ? fJ * cJ * dJ :
84 // max_iter check
85 depth < GCEM_INCML_BETA_MAX_ITER ? // if
86 incomplete_beta_cf(a, b, z, cJ, dJ, fJ * cJ * dJ, depth + 1)
87 :
88 // else
89 fJ * cJ * dJ
90 );
91}

93template <typename T>
94constexpr auto
95incomplete_beta_cf(T const a, T const b, T const z, T const cJ, T const dJ, T const fJ, int const depth) noexcept -> T
96{
97 return incomplete_beta_decision(
98 a,
99 b,
100 z,
101 incomplete_beta_c_update(a, b, z, cJ, depth),
102 incomplete_beta_d_update(a, b, z, dJ, depth),
103 fJ,
104 depth
105 );
106}

108//
109// x^a (1-x)^{b} / (a beta(a,b)) * cf
110
111template <typename T>
112constexpr auto incomplete_beta_begin(T const a, T const b, T const z) noexcept -> T
113{
114 return (
115 (exp(a * log(z) + b * log(T(1) - z) - lbeta(a, b)) / a)
116 * incomplete_beta_cf(
117 a,
118 b,
119 z,
120 T(1),
121 incomplete_beta_d_update(a, b, z, T(1), 0),
122 incomplete_beta_d_update(a, b, z, T(1), 0),
123 1
124 )
125 );
126}

128template <typename T>
129constexpr auto incomplete_beta_check(T const a, T const b, T const z) noexcept -> T
130{
131 return ( // NaN check
132 any_nan(a, b, z) ? etl::numeric_limits<T>::quiet_NaN() :
133 // indistinguishable from zero
134 etl::numeric_limits<T>::epsilon() > z ? T(0)
135 :
136 // parameter check for performance
137 (a + T(1)) / (a + b + T(2)) > z ? incomplete_beta_begin(a, b, z)
138 : T(1) - incomplete_beta_begin(b, a, T(1) - z)
139 );
140}
