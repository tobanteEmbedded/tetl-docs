38 template <typename T>
39 constexpr auto incomplete_beta_coef_even(const T a, const T b, const T z, const int k) noexcept -> T
40 {
41  return (-z * (a + k) * (a + b + k) / ((a + 2 * k) * (a + 2 * k + T(1))));
42 }
43 
44 template <typename T>
45 constexpr auto incomplete_beta_coef_odd(const T a, const T b, const T z, const int k) noexcept -> T
46 {
47  return (z * k * (b - k) / ((a + 2 * k - T(1)) * (a + 2 * k)));
48 }
49 
50 template <typename T>
51 constexpr auto incomplete_beta_coef(const T a, const T b, const T z, const int depth) noexcept -> T
52 {
53  return (!is_odd(depth) ? incomplete_beta_coef_even(a, b, z, depth / 2)
54  : incomplete_beta_coef_odd(a, b, z, (depth + 1) / 2));
55 }

60 template <typename T>
61 constexpr auto incomplete_beta_c_update(const T a, const T b, const T z, const T cJ, const int depth) noexcept -> T
62 {
63  return (T(1) + incomplete_beta_coef(a, b, z, depth) / cJ);

66 template <typename T>
67 constexpr auto incomplete_beta_d_update(const T a, const T b, const T z, const T dJ, const int depth) noexcept -> T
68 {
69  return (T(1) / (T(1) + incomplete_beta_coef(a, b, z, depth) * dJ));

75 template <typename T>
76 constexpr auto incomplete_beta_decision(
77  const T a, const T b, const T z, const T cJ, const T dJ, const T fJ, const int depth) noexcept -> T
78 {
79  return ( // tolerance check
80  abs(cJ * dJ - T(1)) < GCEM_INCML_BETA_TOL ? fJ * cJ * dJ :
81  // max_iter check
82  depth < GCEM_INCML_BETA_MAX_ITER ? // if
83  incomplete_beta_cf(a, b, z, cJ, dJ, fJ * cJ * dJ, depth + 1)
84  :
85  // else
86  fJ * cJ * dJ);
87 }

98 // x^a (1-x)^{b} / (a beta(a,b)) * cf

100 template <typename T>
101 constexpr auto incomplete_beta_begin(const T a, const T b, const T z) noexcept -> T
102 {
103  return ((exp(a * log(z) + b * log(T(1) - z) - lbeta(a, b)) / a)
104  * incomplete_beta_cf(a, b, z, T(1), incomplete_beta_d_update(a, b, z, T(1), 0),
105  incomplete_beta_d_update(a, b, z, T(1), 0), 1));
106 }

108 template <typename T>
109 constexpr auto incomplete_beta_check(const T a, const T b, const T z) noexcept -> T
110 {
111  return ( // NaN check
112  any_nan(a, b, z) ? etl::numeric_limits<T>::quiet_NaN() :
113  // indistinguishable from zero
114  etl::numeric_limits<T>::epsilon() > z ? T(0)
115  :
116  // parameter check for performance
117  (a + T(1)) / (a + b + T(2)) > z ? incomplete_beta_begin(a, b, z)
118  : T(1) - incomplete_beta_begin(b, a, T(1) - z));
119 }
