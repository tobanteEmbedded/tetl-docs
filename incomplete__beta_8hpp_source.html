36 // coefficients; see eq. 18.5.17b

38 template <typename T>
39 constexpr auto incomplete_beta_coef_even(T const a, T const b, T const z, int const k) noexcept -> T
40 {
41  return (-z * (a + k) * (a + b + k) / ((a + 2 * k) * (a + 2 * k + T(1))));
42 }
43 
44 template <typename T>
45 constexpr auto incomplete_beta_coef_odd(T const a, T const b, T const z, int const k) noexcept -> T
46 {
47  return (z * k * (b - k) / ((a + 2 * k - T(1)) * (a + 2 * k)));
48 }
49 
50 template <typename T>
51 constexpr auto incomplete_beta_coef(T const a, T const b, T const z, int const depth) noexcept -> T
52 {
53  return (
54  !is_odd(depth) ? incomplete_beta_coef_even(a, b, z, depth / 2)
55  : incomplete_beta_coef_odd(a, b, z, (depth + 1) / 2)
56  );
57 }

60 // update formulae for the modified Lentz method

62 template <typename T>
63 constexpr auto incomplete_beta_c_update(T const a, T const b, T const z, T const cJ, int const depth) noexcept -> T
64 {
65  return (T(1) + incomplete_beta_coef(a, b, z, depth) / cJ);
66 }
67 
68 template <typename T>
69 constexpr auto incomplete_beta_d_update(T const a, T const b, T const z, T const dJ, int const depth) noexcept -> T
70 {
71  return (T(1) / (T(1) + incomplete_beta_coef(a, b, z, depth) * dJ));
72 }

77 template <typename T>
78 constexpr auto
79 incomplete_beta_decision(T const a, T const b, T const z, T const cJ, T const dJ, T const fJ, int const depth) noexcept
80  -> T
81 {
82  return ( // tolerance check
83  abs(cJ * dJ - T(1)) < GCEM_INCML_BETA_TOL ? fJ * cJ * dJ :
84  // max_iter check
85  depth < GCEM_INCML_BETA_MAX_ITER ? // if
86  incomplete_beta_cf(a, b, z, cJ, dJ, fJ * cJ * dJ, depth + 1)
87  :
88  // else
89  fJ * cJ * dJ
90  );
91 }

108 //
109 // x^a (1-x)^{b} / (a beta(a,b)) * cf
110 
111 template <typename T>
112 constexpr auto incomplete_beta_begin(T const a, T const b, T const z) noexcept -> T
113 {
114  return (
115  (exp(a * log(z) + b * log(T(1) - z) - lbeta(a, b)) / a)
116  * incomplete_beta_cf(
117  a,
118  b,
119  z,
120  T(1),
121  incomplete_beta_d_update(a, b, z, T(1), 0),
122  incomplete_beta_d_update(a, b, z, T(1), 0),
123  1
124  )
125  );
126 }

128 template <typename T>
129 constexpr auto incomplete_beta_check(T const a, T const b, T const z) noexcept -> T
130 {
131  return ( // NaN check
132  any_nan(a, b, z) ? etl::numeric_limits<T>::quiet_NaN() :
133  // indistinguishable from zero
134  etl::numeric_limits<T>::epsilon() > z ? T(0)
135  :
136  // parameter check for performance
137  (a + T(1)) / (a + b + T(2)) > z ? incomplete_beta_begin(a, b, z)
138  : T(1) - incomplete_beta_begin(b, a, T(1) - z)
139  );
140 }

150 /**
151  * Compile-time regularized incomplete beta function
152  *
153  * @param a a real-valued, non-negative input.
154  * @param b a real-valued, non-negative input.
155  * @param z a real-valued, non-negative input.
156  *
157  * @return computes the regularized incomplete beta function,
158  * \f[ \frac{\text{B}(z;\alpha,\beta)}{\text{B}(\alpha,\beta)} =
159  * \frac{1}{\text{B}(\alpha,\beta)}\int_0^z t^{a-1} (1-t)^{\beta-1} dt \f] using
160  * a continued fraction representation, found in the Handbook of Continued
161  * Fractions for Special Functions, and a modified Lentz method. \f[
162  * \frac{\text{B}(z;\alpha,\beta)}{\text{B}(\alpha,\beta)} = \frac{z^{\alpha}
163  * (1-t)^{\beta}}{\alpha \text{B}(\alpha,\beta)} \dfrac{a_1}{1 + \dfrac{a_2}{1 +
164  * \dfrac{a_3}{1 + \dfrac{a_4}{1 + \ddots}}}} \f] where \f$ a_1 = 1 \f$ and \f[
165  * a_{2m+2} = - \frac{(\alpha + m)(\alpha + \beta + m)}{(\alpha + 2m)(\alpha +
166  * 2m + 1)}, \ m \geq 0 \f] \f[ a_{2m+1} = \frac{m(\beta - m)}{(\alpha + 2m -
167  * 1)(\alpha + 2m)}, \ m \geq 1 \f] The Lentz method works as follows: let \f$
168  * f_j \f$ denote the value of the continued fraction up to the first \f$ j \f$
169  * terms; \f$ f_j \f$ is updated as follows: \f[ c_j = 1 + a_j / c_{j-1},
170  * \ \ d_j = 1 / (1 + a_j d_{j-1}) \f] \f[ f_j = c_j d_j f_{j-1} \f]
171  */
172 
173 template <typename T1, typename T2, typename T3>
174 constexpr auto incomplete_beta(const T1 a, const T2 b, const T3 z) noexcept -> common_return_t<T1, T2, T3>
175 {
176  return internal::incomplete_beta_type_check(a, b, z);
177 }
