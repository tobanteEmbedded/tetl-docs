21template <typename Int, bool TerminateWithNull = true>
22[[nodiscard]] constexpr auto from_integer(Int num, char* str, size_t length, int base) -> from_integer_result
23{
24 auto reverseString = [](char* string, etl::size_t len) {
25 etl::size_t f = 0;
26 etl::size_t l = len - 1;
27 while (f < l) {
28 auto const tmp = string[f];
29 string[f] = string[l];
30 string[l] = tmp;
31 f++;
32 l--;
33 }
34 };
35
36 // Handle 0 explicitely, otherwise empty string is printed for 0
37 etl::size_t i = 0;
38 if (num == 0) {
39 if (length < (1 + static_cast<size_t>(TerminateWithNull))) {
40 return {str + length, from_integer_error::overflow};
41 }
42 str[i++] = '0';
43 if constexpr (TerminateWithNull) {
44 str[i] = '\0';
45 }
46 return {&str[i]};
47 }
48
49 bool isNegative = false;
50 if constexpr (is_signed_v<Int>) {
51 if (num < 0 && base == 10) {
52 isNegative = true;
53 num = -num;
54 }
55 }
56
57 while (num != 0) {
58 auto const rem = static_cast<char>(num % static_cast<Int>(base));
59 str[i++] = (rem > 9) ? (rem - 10) + 'a' : rem + '0';
60 num = num / static_cast<Int>(base);
61
62 if (length <= i) {
63 return {nullptr, from_integer_error::overflow};
64 }
65 }
66

69 str[i++] = '-';
70 }
71 }
72

74 str[i] = '\0';
75 }
76
