13 /// \brief Compressed pointer to specified size. Intended to be used as a drop
14 /// in replacement for native pointers.
15 ///
16 /// \details Uses the base address to calculate an offset, which will be stored
17 /// internally. If used on micro controllers, the base address should be set to
18 /// the start of RAM. See your linker script.
19 template <typename Type, intptr_t BaseAddress = 0, typename StorageType = uint16_t>
20 struct small_ptr {
21  /// \brief Default construct empty small_ptr. May contain garbage.
22  small_ptr() = default;
23 
24  /// \brief Construct from nullptr.
25  small_ptr(nullptr_t null) : _value{0} { ignore_unused(null); }
26 
27  /// \brief Construct from raw pointer.
28  small_ptr(Type* ptr) : _value{compress(ptr)} { }
29 
30  /// \brief Returns a raw pointer to Type.
31  [[nodiscard]] auto get() noexcept -> Type* { return reinterpret_cast<Type*>(BaseAddress + _value); }
32 
33  /// \brief Returns a raw pointer to const Type.
34  [[nodiscard]] auto get() const noexcept -> Type const*
35  {
36  return reinterpret_cast<Type const*>(BaseAddress + _value);
37  }
38 
39  /// \brief Returns the compressed underlying integer address.
40  [[nodiscard]] auto compressed_value() const noexcept -> StorageType { return _value; }
41 
42  /// \brief Returns a raw pointer to Type.
43  [[nodiscard]] auto operator->() const -> Type* { return get(); }
44 
45  /// \brief Dereference pointer to Type&.
46  [[nodiscard]] auto operator*() -> Type& { return *get(); }
47 
48  /// \brief Dereference pointer to Type const&.
49  [[nodiscard]] auto operator*() const -> Type const& { return *get(); }
50 
51  /// \brief Pre increment of pointer.
52  [[nodiscard]] auto operator++(int) noexcept -> small_ptr
53  {
54  auto temp = *this;
55  auto* ptr = get();
56  ++ptr;
57  _value = compress(ptr);
58  return temp;
59  }
60 
61  /// \brief Post increment of pointer.
62  [[nodiscard]] auto operator++() noexcept -> small_ptr&
63  {
64  auto* ptr = get();

67  return *this;
68  }

72  {
73  auto temp = *this;
74  auto* ptr = get();

89  /// \brief Returns distance from this to other.
90  [[nodiscard]] auto operator-(small_ptr other) const noexcept -> ptrdiff_t { return get() - other.get(); }
91 
92  /// \brief Implicit conversion to raw pointer to mutable.
93  [[nodiscard]] operator Type*() noexcept { return get(); }
94 
95  /// \brief Implicit conversion to raw pointer to const.
96  [[nodiscard]] operator Type const*() const noexcept { return get(); }
97 
98 private:
99  [[nodiscard]] static auto compress(Type* ptr) -> StorageType
100  {
101  auto const obj = reinterpret_cast<intptr_t>(ptr);
102  return StorageType(obj - BaseAddress);
103  }

106 };
