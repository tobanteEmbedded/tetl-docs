14 /// \brief The struct unique_lock is a general-purpose mutex ownership wrapper
15 /// allowing deferred locking, time-constrained attempts at locking, recursive
16 /// locking, transfer of lock ownership, and use with condition variables.
17 ///
18 /// \details The struct unique_lock is movable, but not copyable -- it meets the
19 /// requirements of MoveConstructible and MoveAssignable but not of
20 /// CopyConstructible or CopyAssignable. The struct unique_lock meets the
21 /// BasicLockable requirements. If Mutex meets the Lockable requirements,
22 /// unique_lock also meets the Lockable requirements (ex.: can be used in lock);
23 /// if Mutex meets the TimedLockable requirements, unique_lock also meets the
24 /// TimedLockable requirements.
25 template <typename Mutex>
26 struct unique_lock {
27 private:
28  Mutex* _mutex{nullptr};
29  bool _owns{false};
30 
31 public:
32  using mutex_type = Mutex;
33 
34  /// \brief Constructs a unique_lock with no associated mutex.
35  unique_lock() noexcept = default;
36 
37  /// \brief Constructs a unique_lock with m as the associated mutex.
38  /// Additionally: Locks the associated mutex by calling m.lock(). The
39  /// behavior is undefined if the current thread already owns the mutex
40  /// except when the mutex is recursive.
41  explicit unique_lock(mutex_type& m) : _mutex{&m} { lock(); }
42 
43  /// \brief Constructs a unique_lock with m as the associated mutex.
44  /// Additionally: Does not lock the associated mutex.
45  unique_lock(mutex_type& m, defer_lock_t /*tag*/) noexcept : _mutex{&m} { }
46 
47  /// \brief Constructs a unique_lock with m as the associated mutex.
48  /// Additionally: Tries to lock the associated mutex without blocking by
49  /// calling m.try_lock(). The behavior is undefined if the current thread
50  /// already owns the mutex except when the mutex is recursive.
51  unique_lock(mutex_type& m, try_to_lock_t /*tag*/) noexcept : _mutex{&m} { try_lock(); }
52 
53  /// \brief Constructs a unique_lock with m as the associated mutex.
54  /// Additionally: Assumes the calling thread already owns m.
55  unique_lock(mutex_type& m, adopt_lock_t /*tag*/) : _mutex{&m}, _owns{true} { }
56 
57  /// \brief Constructs a unique_lock with m as the associated mutex.
58  /// Additionally: Tries to lock the associated mutex by calling
59  /// m.try_lock_until(timeout_time). Blocks until specified timeout_time has
60  /// been reached or the lock is acquired, whichever comes first. May block
61  /// for longer than until timeout_time has been reached.
62  template <typename Clock, typename Duration>
63  unique_lock(mutex_type& m, chrono::time_point<Clock, Duration> const& absTime) noexcept : _mutex{&m}
64  {
65  try_lock_until(absTime);
66  }
67 
68  /// \brief Constructs a unique_lock with m as the associated mutex.
69  /// Additionally: Tries to lock the associated mutex by calling
70  /// m.try_lock_for(timeout_duration). Blocks until specified
71  /// timeout_duration has elapsed or the lock is acquired, whichever comes
72  /// first. May block for longer than timeout_duration.
73  template <typename Rep, typename Period>
74  unique_lock(mutex_type& m, chrono::duration<Rep, Period> const& relTime) noexcept : _mutex{&m}
75  {
76  try_lock_for(relTime);
77  }
78 
79  /// \brief Deleted copy constructor. unique_lock is move only.
80  unique_lock(unique_lock const&) = delete;
81 
82  /// \brief Deleted copy assignment. unique_lock is move only.
83  auto operator=(unique_lock const&) -> unique_lock& = delete;
84 
85  /// \brief Move constructor. Initializes the unique_lock with the contents
86  /// of other. Leaves other with no associated mutex.
87  unique_lock(unique_lock&& u) noexcept : _mutex{exchange(u._mutex, nullptr)}, _owns{exchange(u._owns, false)} { }
88 
89  /// \brief Move assignment operator. Replaces the contents with those of
90  /// other using move semantics. If prior to the call *this has an associated
91  /// mutex and has acquired ownership of it, the mutex is unlocked.
92  auto operator=(unique_lock&& u) noexcept -> unique_lock&
93  {
94  if (_mutex != nullptr && _owns) {
95  unlock();
96  }
97  _mutex = exchange(u._mutex, nullptr);
98  _owns = exchange(u._owns, false);
99  return *this;
100  }
101 
102  ~unique_lock() noexcept { unlock(); }
103 
104  /// \brief Locks (i.e., takes ownership of) the associated mutex.
105  auto lock() noexcept(noexcept(_mutex->lock())) -> void
106  {
107  if ((_mutex != nullptr) and !_owns) {
108  _mutex->lock();
109  _owns = true;
110  }
111  }
112 
113  /// \brief Tries to lock (i.e., takes ownership of) the associated mutex
114  /// without blocking.
115  /// \returns true if the ownership of the mutex has been acquired
116  /// successfully, false otherwise.
117  auto try_lock() noexcept(noexcept(_mutex->try_lock())) -> bool
118  {
119  if ((_mutex != nullptr) && !_owns) {
120  if (auto success = _mutex->try_lock(); success) {
121  _owns = true;
122  return true;
123  }
124  }
125 
126  return false;
127  }
128 
129  /// \brief Tries to lock (i.e., takes ownership of) the associated mutex.
130  /// Blocks until specified timeout_duration has elapsed or the lock is
131  /// acquired, whichever comes first. On successful lock acquisition returns
132  /// true, otherwise returns false. Effectively calls
133  /// mutex()->try_lock_for(timeout_duration). This function may block for
134  /// longer than timeout_duration due to scheduling or resource contention
135  /// delays.
136  template <typename Rep, typename Period>
137  auto try_lock_for(chrono::duration<Rep, Period> const& dur) noexcept(noexcept(_mutex->try_lock_for(dur))) -> bool
138  {
139  if ((_mutex != nullptr) && !_owns) {
140  if (auto success = _mutex->try_lock_for(dur); success) {
141  _owns = true;
142  return true;
143  }
144  }
145  return false;
146  }
147 
148  /// \brief Tries to lock (i.e., takes ownership of) the associated mutex
149  /// without blocking.
150  template <typename Clock, typename Duration>
151  auto try_lock_until(chrono::time_point<Clock, Duration> const& tp) noexcept(noexcept(_mutex->try_lock_until(tp))
152  ) -> bool
153  {
154  if ((_mutex != nullptr) && !_owns) {
155  if (auto success = _mutex->try_lock_until(tp); success) {
156  _owns = true;
157  return true;
158  }
159  }
160  return false;
161  }
162 
163  /// \brief Unlocks (i.e., releases ownership of) the associated mutex and
164  /// releases ownership. Silently does nothing, if there is no associated
165  /// mutex or if the mutex is not locked.
166  auto unlock() -> void
167  {
168  if ((_mutex != nullptr) and _owns) {
169  _mutex->unlock();
170  _owns = false;
171  }
172  }
173 
174  /// \brief Exchanges the internal states of the lock objects.
175  auto swap(unique_lock& other) noexcept -> void
176  {
177  using etl::swap;
178  swap(_mutex, other._mutex);
179  swap(_owns, other._owns);
180  }
181 
182  /// \brief Breaks the association of the associated mutex, if any, and
183  /// *this. No locks are unlocked. If the *this held ownership of the
184  /// associated mutex prior to the call, the caller is now responsible to
185  /// unlock the mutex.
186  ///
187  /// \returns Pointer to the associated mutex or a null pointer if there was
188  /// no associated mutex.
189  [[nodiscard]] auto release() noexcept -> mutex_type*
190  {
191  _owns = false;
192  return _mutex;
193  }
194 
195  /// \brief Checks whether *this owns a locked mutex or not.
196  [[nodiscard]] auto owns_lock() const noexcept -> bool { return _owns; }
197 
198  /// \brief Checks whether *this owns a locked mutex or not.
199  [[nodiscard]] explicit operator bool() const noexcept { return owns_lock(); }
200 
201  /// \brief Returns a pointer to the associated mutex, or a null pointer if
202  /// there is no associated mutex.
203  [[nodiscard]] auto mutex() const noexcept -> mutex_type* { return _mutex; }
204 };

206 /// \brief Specializes the swap algorithm for unique_lock. Exchanges the state
207 /// of lhs with that of rhs.
208 template <typename Mutex>
209 void swap(unique_lock<Mutex>& lhs, unique_lock<Mutex>& rhs) noexcept(noexcept(lhs.swap(rhs)))
210 {
211  lhs.swap(rhs);
