11 /// \brief front_insert_iterator is an LegacyOutputIterator that prepends
12 /// elements to a container for which it was constructed. The container's
13 /// push_front() member function is called whenever the iterator (whether
14 /// dereferenced or not) is assigned to. Incrementing the
15 /// front_insert_iterator is a no-op.
16 ///
17 /// \todo Add tests when a container with push_front has been implemented.
18 template <typename Container>
19 struct front_insert_iterator {
20 protected:
21  Container* container_ = nullptr;
22 
23 public:
24  using iterator_category = output_iterator_tag;
25  using value_type = void;
26  using difference_type = void;
27  using pointer = void;
28  using reference = void;
29  using container_type = Container;
30 
31  /// \brief Initializes the underlying pointer to container with nullptr.
32  constexpr front_insert_iterator() noexcept = default;
33 
34  /// \brief Initializes the underlying pointer to the container to
35  /// addressof(c).
36  constexpr explicit front_insert_iterator(Container& container) : container_{addressof(container)} { }
37 
38  /// \brief Inserts the given value value to the container.
39  constexpr auto operator=(typename Container::value_type const& value) -> front_insert_iterator&
40  {
41  container_->push_front(value);
42  return *this;
43  }
44 
45  /// \brief Inserts the given value value to the container.
46  constexpr auto operator=(typename Container::value_type&& value) -> front_insert_iterator&
47  {
48  container_->push_front(TETL_MOVE(value));
49  return *this;
50  }
51 
52  /// \brief Does nothing, this member function is provided to satisfy the
53  /// requirements of LegacyOutputIterator. It returns the iterator itself,
54  /// which makes it possible to use code such as *iter = value to output
55  /// (insert) the value into the underlying container.
56  constexpr auto operator*() -> front_insert_iterator& { return *this; }
57 
58  /// \brief Does nothing. These operator overloads are provided to satisfy
