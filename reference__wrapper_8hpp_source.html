28 /// \brief reference_wrapper is a class template that wraps a reference in a
29 /// copyable, assignable object. It is frequently used as a mechanism to store
30 /// references inside standard containers (like etl::static_vector) which cannot
31 /// normally hold references. Specifically, reference_wrapper is a
32 /// CopyConstructible and CopyAssignable wrapper around a reference to object or
33 /// reference to function of type T. Instances of reference_wrapper are objects
34 /// (they can be copied or stored in containers) but they are implicitly
35 /// convertible to T&, so that they can be used as arguments with the functions
36 /// that take the underlying type by reference. If the stored reference is
37 /// Callable, reference_wrapper is callable with the same arguments.
38 template <typename T>
39 struct reference_wrapper {
40  using type = T;
41 
42  /// \brief Constructs a new reference wrapper. Converts x to T& as if by T&
43  /// t = TETL_FORWARD(x);, then stores a reference to t. This overload only
44  /// participates in overload resolution if `decay_t<U>` is not the same type
45  /// as reference_wrapper and the expression `FUN(declval<U>())` is
46  /// well-formed, where FUN names the set of imaginary functions:
47  ///
48  /// \code
49  /// void FUN(T&) noexcept;
50  /// void FUN(T&&) = delete;
51  /// \endcode
52  ///
53  /// https://en.cppreference.com/w/cpp/utility/functional/reference_wrapper/reference_wrapper
54  template <
55  typename U,
56  typename
57  = decltype(detail::FUN<T>(declval<U>()), enable_if_t<!is_same_v<reference_wrapper, remove_cvref_t<U>>>())>
58  constexpr reference_wrapper(U&& u) noexcept(noexcept(detail::FUN<T>(TETL_FORWARD(u))))
59  : _ptr(addressof(detail::FUN<T>(TETL_FORWARD(u))))
60  {
61  }
62 
63  /// \brief Constructs a new reference wrapper. Copy constructor. Stores a
64  /// reference to other.get().
65  constexpr reference_wrapper(reference_wrapper const& x) noexcept = default;
66 
67  /// \brief Copy assignment operator. Drops the current reference and stores
68  /// a reference to other.get().
69  constexpr auto operator=(reference_wrapper const& x) noexcept -> reference_wrapper& = default;
70 
71  /// \brief Returns the stored reference.
72  [[nodiscard]] constexpr operator type&() const noexcept { return *_ptr; }
73 
74  /// \brief Returns the stored reference.
75  [[nodiscard]] constexpr auto get() const noexcept -> type& { return *_ptr; }
76 
77  /// \brief Calls the Callable object, reference to which is stored. This
78  /// function is available only if the stored reference points to a Callable
79  /// object. T must be a complete type.
80  ///
81  /// \returns The return value of the called function.
82  template <typename... Args>
83  constexpr auto operator()(Args&&... args) const
84  noexcept(noexcept(invoke(get(), TETL_FORWARD(args)...))) -> invoke_result_t<T&, Args...>
85  {
86  return invoke(get(), TETL_FORWARD(args)...);
87  }
88 
89 private:
90  type* _ptr;
91 };
