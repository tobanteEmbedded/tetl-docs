181template <typename T>
182template <typename X>
183constexpr auto complex<T>::operator/=(complex<X> const& val) -> complex<T>&
184{
185 auto const norm = [](auto const& c) {
186 auto const x = c.real();
187 auto const y = c.imag();
188 return static_cast<T>(x * x + y * y);
189 };
190
191 auto const r = static_cast<T>(_real * val.real() + _imag * val.imag());
192 auto const n = norm(val);
193 _imag = (_imag * val.real() - _real * val.imag()) / n;
194 _real = r / n;
195 return *this;
196}
