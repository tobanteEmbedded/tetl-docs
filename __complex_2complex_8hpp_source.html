178{

185template <typename T>
186template <typename X>
187constexpr auto complex<T>::operator/=(complex<X> const& val) -> complex<T>&
188{
189 auto const norm = [](auto const& c) {
190 auto const x = c.real();
191 auto const y = c.imag();
192 return static_cast<T>(x * x + y * y);
193 };
194
195 auto const r = static_cast<T>(_real * val.real() + _imag * val.imag());
196 auto const n = norm(val);
197 _imag = (_imag * val.real() - _real * val.imag()) / n;
198 _real = r / n;
199 return *this;
200}
