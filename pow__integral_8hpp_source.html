33// integral-valued powers using method described in
34// https://en.wikipedia.org/wiki/Exponentiation_by_squaring
35
36template <typename T1, typename T2>
37constexpr auto pow_integral_compute_recur(const T1 base, const T1 val, const T2 expTerm) noexcept -> T1
38{
39 return (
40 expTerm > T2(1) ? (is_odd(expTerm) ? pow_integral_compute_recur(base * base, val * base, expTerm / 2)
41 : pow_integral_compute_recur(base * base, val, expTerm / 2))
42 : (expTerm == T2(1) ? val * base : val)
43 );
44}

46template <typename T1, typename T2, typename etl::enable_if<etl::is_signed<T2>::value>::type* = nullptr>
47constexpr auto pow_integral_sgn_check(const T1 base, const T2 expTerm) noexcept -> T1
48{
49 return (
50 expTerm < T2(0) ? //
51 T1(1) / pow_integral_compute(base, -expTerm)
52 :
53 //
54 pow_integral_compute_recur(base, T1(1), expTerm)
55 );
56}

58template <typename T1, typename T2, typename etl::enable_if<!etl::is_signed<T2>::value>::type* = nullptr>
59constexpr auto pow_integral_sgn_check(const T1 base, const T2 expTerm) noexcept -> T1
60{
61 return (pow_integral_compute_recur(base, T1(1), expTerm));
62}

64template <typename T1, typename T2>
65constexpr auto pow_integral_compute(const T1 base, const T2 expTerm) noexcept -> T1
66{
67 return (
68 expTerm == T2(3) ? base * base * base
69 : expTerm == T2(2) ? base * base
70 : expTerm == T2(1) ? base
71 : expTerm == T2(0) ? T1(1)
72 :
73 // check for overflow
74 expTerm == etl::numeric_limits<T2>::min() ? T1(0)
75 : expTerm == etl::numeric_limits<T2>::max() ? etl::numeric_limits<T1>::infinity()
76 :
77 // else
78 pow_integral_sgn_check(base, expTerm)
79 );
80}
