125 template <typename T, bool = etl::is_reference_v<T>>
126 struct optional_storage_base : optional_destruct_base<T> {
127  using base_t = optional_destruct_base<T>;
128  using value_type = T;
129  using base_t::base_t;
130 
131  [[nodiscard]] constexpr auto has_value() const noexcept -> bool { return this->internal_has_value; }
132 
133  [[nodiscard]] constexpr auto get() & noexcept -> value_type& { return this->internal_value; }
134 
135  [[nodiscard]] constexpr auto get() const& noexcept -> value_type const& { return this->internal_value; }
136 
137  [[nodiscard]] constexpr auto get() && noexcept -> value_type&& { return TETL_MOVE(this->internal_value); }
138 
139  [[nodiscard]] constexpr auto get() const&& noexcept -> value_type const&&
140  {
141  return TETL_MOVE(this->internal_value);
142  }
143 
144  template <typename... Args>
145 
146  void construct(Args&&... args)
147  {
148  ::new (static_cast<void*>(etl::addressof(this->internal_value))) value_type(TETL_FORWARD(args)...);
149  this->internal_has_value = true;
150  }
151 
152  template <typename U>
153  void construct_from(U&& opt)
154  {
155  if (opt.has_value()) {
156  construct(TETL_FORWARD(opt).get());
157  }
158  }
159 
160  template <typename U>
161  void assign_from(U&& opt)
162  {
163  if (this->internal_has_value == opt.has_value()) {
164  if (this->internal_has_value) {
165  this->internal_value = TETL_FORWARD(opt).get();
166  }
167  } else {
168  if (this->internal_has_value) {
169  this->reset();
170  } else {
171  construct(TETL_FORWARD(opt).get());
172  }
173  }
174  }

290 /// \brief The class template optional manages an optional contained value,
291 /// i.e. a value that may or may not be present.
292 ///
293 /// \details A common use case for optional is the return value of a function
294 /// that may fail. As opposed to other approaches, such as etl::pair<T,bool>,
295 /// optional handles expensive-to-construct objects well and is more readable,
296 /// as the intent is expressed explicitly.
297 ///
298 /// Any instance of optional at any given point in time either contains a
299 /// value or does not contain a value.
300 ///
301 /// If an optional contains a value, the value is guaranteed to be
302 /// allocated as part of the optional object footprint, i.e. no dynamic memory
303 /// allocation ever takes place. Thus, an optional object models an object, not
304 /// a pointer, even though operator*() and operator->() are defined.
305 ///
306 /// When an object of type optional is contextually converted to bool, the
307 /// conversion returns true if the object contains a value and false if it does
308 /// not contain a value.
309 ///
310 /// The optional object contains a value in the following conditions:
311 /// - The object is initialized with/assigned from a value of type T or
312 /// another optional that contains a value.
313 ///
314 /// The object does not contain a value in the following conditions:
315 /// - The object is default-initialized.
316 /// - The object is initialized with/assigned from a value of type
317 /// etl::nullopt_t or an optional object that does not contain a value.
318 /// - The member function reset() is called.
319 ///
320 /// There are no optional references; a program is ill-formed if it instantiates
321 /// an optional with a reference type. Alternatively, an optional of a
322 /// etl::reference_wrapper of type T may be used to hold a reference. In
323 /// addition, a program is ill-formed if it instantiates an optional with the
324 /// (possibly cv-qualified) tag types etl::nullopt_t or etl::in_place_t.
325 ///
326 /// https://en.cppreference.com/w/cpp/utility/optional
327 ///
328 /// \tparam T The type of the value to manage initialization state for. The type
329 /// must meet the requirements of Destructible (in particular, array types are
330 /// not allowed).
331 ///
332 /// \headerfile etl/optional.hpp
333 /// \include optional.cpp

760  /// \brief Swaps the contents with those of other.
761  constexpr auto swap(optional& other
762  ) noexcept(etl::is_nothrow_move_constructible_v<value_type> && etl::is_nothrow_swappable_v<value_type>) -> void
763  {
764  // If neither *this nor other contain a value, the function has no
765  // effect.
766 
767  // If both *this and other contain values, the contained values are
768  // exchanged
769  if (this->has_value() == other.has_value()) {
770  using etl::swap;
771  if (this->has_value()) {
772  swap(this->get(), other.get());
773  }
774  return;
775  }
776 
777  // If only one of *this and other contains a value (let's call this
778  // object in and the other un), the contained value of un is
779  // direct-initialized from TETL_MOVE(*in), followed by destruction of
780  // the contained value of in as if by in->T::~T(). After this call, in
781  // does not contain a value; un contains a value.
782  if (this->has_value()) {
783  other.construct(TETL_MOVE(this->get()));
784  reset();
785  return;
786  }
787 
788  this->construct(TETL_MOVE(other.get()));
789  other.reset();
790  }

1292 /// \brief The template specialization of etl::hash for the etl::optional class
1293 /// allows users to obtain hashes of the values contained in optional objects.
1294 ///
1295 /// \details The specialization etl::hash<optional<T>> is enabled (see
1296 /// etl::hash) if etl::hash<etl::remove_const_t<T>> is enabled, and is disabled
1297 /// otherwise.
1298 ///
1299 /// When enabled, for an object opt of type etl::optional<T> that contains a
1300 /// value, etl::hash<etl::optional<T>>()(opt) evaluates to the same value as
1301 /// etl::hash<etl::remove_const_t<T>>()(*opt). For an optional that does not
1302 /// contain a value, the hash is unspecified.
1303 ///
1304 /// The member functions of this specialization are not guaranteed to be
1305 /// noexcept because the hash of the underlying type might throw.
1306 ///
1307 /// https://en.cppreference.com/w/cpp/utility/optional/hash
1308 ///
1309 /// \headerfile etl/optional.hpp
1310 template <typename T>
1311 struct hash<etl::optional<T>> {
1312  [[nodiscard]] constexpr auto operator()(etl::optional<T> const& opt) const -> etl::size_t
1313  {
1314  using type = etl::remove_const_t<T>;
1315  static_assert(etl::is_specialized_v<etl::hash, type>);
1316  return static_cast<bool>(opt) ? etl::hash<type>{}(*opt) : 0;
1317  }
1318 };
