60template <typename T>
61struct optional_destruct_base<T, false> {
62 using value_type = T;
63 static_assert(etl::is_object_v<value_type>, "undefined behavior");
64
65 ~optional_destruct_base()
66 {
67 if (internal_has_value) {
68 internal_value.~value_type();
69 }
70 }
71
72 constexpr optional_destruct_base() noexcept { }
73
74 template <typename... Args>
75 constexpr explicit optional_destruct_base(etl::in_place_t /*tag*/, Args&&... args)
76 : internal_value(TETL_FORWARD(args)...)
77 , internal_has_value(true)
78 {
79 }
80
81 auto reset() noexcept -> void
82 {
83 if (internal_has_value) {
84 internal_value.~value_type();
85 internal_has_value = false;
86 }
87 }
88
89 union {
90 char internal_null_state{};
91 value_type internal_value;
92 };
93
94 bool internal_has_value = false;
95};

126template <typename T, bool = etl::is_reference_v<T>>
127struct optional_storage_base : optional_destruct_base<T> {
128 using base_t = optional_destruct_base<T>;
129 using value_type = T;
130 using base_t::base_t;
131
132 [[nodiscard]] constexpr auto has_value() const noexcept -> bool { return this->internal_has_value; }
133
134 [[nodiscard]] constexpr auto get() & noexcept -> value_type& { return this->internal_value; }
135
136 [[nodiscard]] constexpr auto get() const& noexcept -> value_type const& { return this->internal_value; }
137
138 [[nodiscard]] constexpr auto get() && noexcept -> value_type&& { return TETL_MOVE(this->internal_value); }
139
140 [[nodiscard]] constexpr auto get() const&& noexcept -> value_type const&&
141 {
142 return TETL_MOVE(this->internal_value);
143 }
144
145 template <typename... Args>
146
147 void construct(Args&&... args)
148 {
149 ::new (static_cast<void*>(etl::addressof(this->internal_value))) value_type(TETL_FORWARD(args)...);
150 this->internal_has_value = true;
151 }
152
153 template <typename U>
154 void construct_from(U&& opt)
155 {
156 if (opt.has_value()) {
157 construct(TETL_FORWARD(opt).get());
158 }
159 }
160
161 template <typename U>
162 void assign_from(U&& opt)
163 {
164 if (this->internal_has_value == opt.has_value()) {
165 if (this->internal_has_value) {
166 this->internal_value = TETL_FORWARD(opt).get();
167 }
168 } else {
169 if (this->internal_has_value) {
170 this->reset();
171 } else {
172 construct(TETL_FORWARD(opt).get());
173 }
174 }
175 }

762 constexpr auto swap(optional& other
763 ) noexcept(etl::is_nothrow_move_constructible_v<value_type> && etl::is_nothrow_swappable_v<value_type>) -> void
764 {
765 // If neither *this nor other contain a value, the function has no
766 // effect.
767
768 // If both *this and other contain values, the contained values are
769 // exchanged
770 if (this->has_value() == other.has_value()) {
771 using etl::swap;
772 if (this->has_value()) {
773 swap(this->get(), other.get());
774 }
775 return;
776 }
777
778 // If only one of *this and other contains a value (let's call this
779 // object in and the other un), the contained value of un is
780 // direct-initialized from TETL_MOVE(*in), followed by destruction of
781 // the contained value of in as if by in->T::~T(). After this call, in
782 // does not contain a value; un contains a value.
783 if (this->has_value()) {
784 other.construct(TETL_MOVE(this->get()));
785 reset();
786 return;
787 }
788
789 this->construct(TETL_MOVE(other.get()));
790 other.reset();
791 }
