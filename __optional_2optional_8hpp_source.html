479struct optional<T&> {

490 requires(not etl::is_same_v<etl::remove_cvref_t<U>, optional>)
491 constexpr explicit(not etl::is_convertible_v<U, T>) optional(U&& v)
492 : _ptr(etl::addressof(v))
493 {
494 static_assert(etl::is_constructible_v<etl::add_lvalue_reference_t<T>, U>, "Must be able to bind U to T&");
495 static_assert(etl::is_lvalue_reference_v<U>, "U must be an lvalue");
496 }

498 template <typename U>
499 requires(not etl::is_same_v<etl::remove_cvref_t<U>, optional>)
500 constexpr explicit(not etl::is_convertible_v<U, T>) optional(optional<U> const& rhs)
501 : _ptr(etl::addressof(*rhs))
502 {
503 }
504
505 constexpr optional(optional const& other) = default;
506 constexpr optional(optional&& other) noexcept = default;
507 constexpr ~optional() = default;
508
509 constexpr auto operator=(optional const&) noexcept -> optional& = default;
510 constexpr auto operator=(optional&&) noexcept -> optional& = default;
511
512 constexpr auto operator=(etl::nullopt_t /*tag*/) noexcept -> optional&
513 {
514 _ptr = nullptr;
515 return *this;
516 }
517
518 template <typename U = T>
519 requires(not etl::is_same_v<etl::remove_cvref_t<U>, optional>
520 and not etl::conjunction_v<etl::is_scalar<T>, etl::is_same<T, etl::decay_t<U>>>)
521 constexpr auto operator=(U&& v) -> optional&
522 {
523 static_assert(etl::is_constructible_v<etl::add_lvalue_reference_t<T>, U>, "Must be able to bind U to T&");
524 static_assert(etl::is_lvalue_reference_v<U>, "U must be an lvalue");
525 _ptr = etl::addressof(v);
526 return *this;
527 }
528
529 template <typename U>
530 constexpr auto operator=(optional<U> const& rhs) -> optional&
531 {
532 static_assert(etl::is_constructible_v<etl::add_lvalue_reference_t<T>, U>, "Must be able to bind U to T&");
533 _ptr = rhs._ptr;
534 return *this;
535 }
536
537 template <typename U = T>
538 requires(not etl::is_same_v<remove_cvref_t<U>, optional>)
539 constexpr auto emplace(U&& u) noexcept -> optional&
540 {
541 *this = etl::forward<U>(u);
542 return *this;
543 }
544
545 [[nodiscard]] constexpr auto operator->() const noexcept -> T* { return _ptr; }
546
547 [[nodiscard]] constexpr auto operator*() const noexcept -> T& { return *_ptr; }
548
549 [[nodiscard]] constexpr explicit operator bool() const noexcept { return has_value(); }
550
551 [[nodiscard]] constexpr auto has_value() const noexcept -> bool { return _ptr != nullptr; }
552
553 constexpr void reset() noexcept { _ptr = nullptr; }
554
555 constexpr void swap(optional& rhs) noexcept { etl::swap(_ptr, rhs._ptr); }
556
557private:
558 T* _ptr{nullptr};
559};
