16[[nodiscard]] constexpr auto last_day_of_month(chrono::year const& y, chrono::month const& m) -> chrono::day
17{
18 constexpr chrono::day lastDays[] = {
19 chrono::day{31},
20 chrono::day{28},
21 chrono::day{31},
22 chrono::day{30},
23 chrono::day{31},
24 chrono::day{30},
25 chrono::day{31},
26 chrono::day{31},
27 chrono::day{30},
28 chrono::day{31},
29 chrono::day{30},
30 chrono::day{31},
31 };
32
33 if (m == chrono::month{2} and y.is_leap()) {
34 return chrono::day{29};
