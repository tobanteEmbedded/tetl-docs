25 template <etl::size_t Offset, etl::size_t Count, etl::size_t Extent>
26 [[nodiscard]] consteval auto subspan_extent() -> etl::size_t
27 {
28  if (Count != dynamic_extent) {
29  return Count;
30  }
31  if (Extent != dynamic_extent) {
32  return Extent - Offset;
33  }
34  return dynamic_extent;
35 }

38 /// \brief A non-owning view over a contiguous sequence of objects.
39 ///
40 /// \details The class template span describes an object that can refer to a
41 /// contiguous sequence of objects with the first element of the sequence at
42 /// position zero. A span can either have a static extent, in which case the
43 /// number of elements in the sequence is known and encoded in the type, or a
44 /// dynamic extent.
45 ///
46 /// If a span has dynamic extent a typical implementation holds
47 /// two members: a pointer to T and a size. A span with static extent may have
48 /// only one member: a pointer to T.
49 template <typename ElementType, size_t Extent = etl::dynamic_extent>
50 struct span {
51  using element_type = ElementType;
52  using value_type = etl::remove_cv_t<ElementType>;
53  using size_type = etl::size_t;
54  using difference_type = etl::ptrdiff_t;
55  using pointer = ElementType*;
56  using const_pointer = ElementType const*;
57  using reference = ElementType&;
58  using const_reference = ElementType const&;
59  using iterator = ElementType*;
60  using reverse_iterator = etl::reverse_iterator<iterator>;
61 
62  /// \brief The number of elements in the sequence, or etl::dynamic_extent
63  /// if dynamic.
64  static constexpr size_type extent = Extent;
65 
66  /// \brief Constructs a span. Constructs an empty span whose
67  /// data() == nullptr and size() == 0.
68  ///
69  /// \details This overload only participates in overload resolution
70  /// if extent == 0 || extent == etl::dynamic_extent.
71  constexpr span() noexcept
72  requires(extent == 0 or extent == dynamic_extent)
73  = default;
74 
75  /// \brief Constructs a span.
76  ///
77  /// \details Constructs a span that is a view over the range [first, first +
78  /// count); the resulting span has data() == etl::to_address(first) and
79  /// size()
80  /// == count. The behavior is undefined if [first, first + count) is not a
81  /// valid range, if It does not actually model contiguous_iterator, or if
82  /// extent != etl::dynamic_extent && count != extent. This overload only
83  /// participates in overload resolution if, It satisfies contiguous_iterator
84  /// and the conversion from etl::iter_reference_t<It> to element_type is at
85  /// most a qualification conversion.
86  template <typename It>
87  explicit(extent != dynamic_extent) constexpr span(It first, size_type count) : _data{first}
88  , _size{count}
89  {
90  }
91 
92  // /// \brief Constructs a span.
93  // template <typename It, typename End>
94  // explicit(extent != etl::dynamic_extent) constexpr span(It first, End last);
95 
96  /// \brief Constructs a span. From a c style array.
97  template <size_t N>
98  constexpr span(element_type (&arr)[N]) noexcept : _data{&arr[0]}
99  , _size{N}
100  {
101  }
102 
103  /// \brief Constructs a span. From a array<Type,Size>.
104  template <typename U, size_t N>
105  constexpr span(array<U, N>& arr) noexcept : _data{arr.data()}
106  , _size{arr.size()}
107  {
108  }
109 
110  /// \brief Constructs a span. From a array<Type,Size> const.
111  template <typename U, size_t N>
112  constexpr span(array<U, N> const& arr) noexcept : _data{arr.data()}
113  , _size{arr.size()}
114  {
115  }
116 
117  /// \brief Constructs a span.
118  template <typename R>
119  explicit(extent != dynamic_extent) constexpr span(R&& r) : _data{r.data()}
120  , _size{r.size()}
121  {
122  }
123 
124  // /// \brief Constructs a span.
125  // template <typename U, etl::size_t N>
126  // explicit(extent != etl::dynamic_extent
127  // && N == etl::dynamic_extent) constexpr span( etl::span<U,
128  // N> const& s) noexcept;

130  /// \brief Constructs a span.
131  constexpr span(span const& other) noexcept = default;
132 
133  /// \brief Returns an iterator to the first element of the span. If the span
134  /// is empty, the returned iterator will be equal to end().
135  [[nodiscard]] constexpr auto begin() const noexcept -> iterator { return &_data[0]; }
136 
137  /// \brief Returns an iterator to the element following the last element of
138  /// the span. This element acts as a placeholder; attempting to access it
139  /// results in undefined behavior
140  [[nodiscard]] constexpr auto end() const noexcept -> iterator { return begin() + size(); }
141 
142  /// \brief Returns a reverse iterator to the first element of the reversed
143  /// span. It corresponds to the last element of the non-reversed span. If
144  /// the span is empty, the returned iterator is equal to rend().
145  [[nodiscard]] auto rbegin() const noexcept -> reverse_iterator { return reverse_iterator(end()); }
146 
147  /// \brief Returns a reverse iterator to the element following the last
148  /// element of the reversed span. It corresponds to the element preceding
149  /// the first element of the non-reversed span. This element acts as a
150  /// placeholder, attempting to access it results in undefined behavior.
151  [[nodiscard]] auto rend() const noexcept -> reverse_iterator { return reverse_iterator(begin()); }
152 
153  /// \brief Returns a reference to the first element in the span. Calling
154  /// front on an empty span results in undefined behavior.
155  [[nodiscard]] constexpr auto front() const -> reference { return *begin(); }
156 
157  /// \brief Returns a reference to the last element in the span. Calling
158  /// front on an empty span results in undefined behavior.
159  [[nodiscard]] constexpr auto back() const -> reference { return *(end() - 1); }
160 
161  /// \brief Returns a reference to the idx-th element of the sequence. The
162  /// behavior is undefined if idx is out of range (i.e., if it is greater
163  /// than or equal to size()).
164  [[nodiscard]] constexpr auto operator[](size_type idx) const -> reference { return data()[idx]; }
165 
166  /// \brief Returns a pointer to the beginning of the sequence.
167  [[nodiscard]] constexpr auto data() const noexcept -> pointer { return _data; }
168 
169  /// \brief Returns the number of elements in the span.
170  [[nodiscard]] constexpr auto size() const noexcept -> size_type { return _size; }
171 
172  /// \brief Returns the number of elements in the span.
173  [[nodiscard]] constexpr auto size_bytes() const noexcept -> size_type { return size() * sizeof(element_type); }
174 
175  /// \brief Checks if the span is empty.
176  [[nodiscard]] constexpr auto empty() const noexcept -> bool { return size() == 0; }
177 
178  /// \brief Obtains a span that is a view over the first Count elements of
179  /// this span. The program is ill-formed if Count > Extent.
180  template <size_t Count>
181  [[nodiscard]] constexpr auto first() const -> span<element_type, Count>
182  {
183  static_assert(!(Count > Extent));
184  return span<element_type, Count>{data(), static_cast<size_type>(Count)};
185  }
186 
187  /// \brief Obtains a span that is a view over the first Count elements of
188  /// this span. The behavior is undefined if Count > size().
189  [[nodiscard]] constexpr auto first(size_type count) const -> span<element_type, dynamic_extent>
190  {
191  TETL_ASSERT(!(count > size()));
192  return {data(), static_cast<size_type>(count)};
193  }
194 
195  /// \brief Obtains a span that is a view over the last Count elements of
196  /// this span. The program is ill-formed if Count > Extent.
197  template <size_t Count>
198  [[nodiscard]] constexpr auto last() const -> span<element_type, Count>
199  {
200  static_assert(!(Count > Extent));
201  return span<element_type, Count>{data() + (size() - Count), static_cast<size_type>(Count)};
202  }
203 
204  /// \brief Obtains a span that is a view over the last Count elements of
205  /// this span. The behavior is undefined if Count > size().
206  [[nodiscard]] constexpr auto last(size_type count) const -> span<element_type, dynamic_extent>
207  {
208  TETL_ASSERT(!(count > size()));
209  return {data() + (size() - count), static_cast<size_type>(count)};
210  }
211 
212  /// \brief Obtains a span that is a view over the Count elements of this
213  /// span starting at offset Offset. If Count is etl::dynamic_extent, the
214  /// number of elements in the subspan is size() - offset (i.e., it ends at
215  /// the end of *this.).
216  template <size_t Offset, size_t Count = dynamic_extent>
217  [[nodiscard]] constexpr auto subspan() const -> span<element_type, detail::subspan_extent<Offset, Count, Extent>()>
218  {
219  static_assert(!(Offset > Extent));
220  static_assert(!(Count == dynamic_extent || Count <= Extent - Offset)); // NOLINT(*-simplify-boolean-expr)
221  auto const sz = Count == dynamic_extent ? size() - Offset : Count;
222  return {data() + Offset, static_cast<size_type>(sz)};
223  }
224 
225  /// \brief Obtains a span that is a view over the Count elements of this
226  /// span starting at offset Offset. If Count is etl::dynamic_extent, the
227  /// number of elements in the subspan is size() - offset (i.e., it ends at
228  /// the end of *this.).
229  [[nodiscard]] constexpr auto
230  subspan(size_type offset, size_type count = dynamic_extent) const -> span<element_type, dynamic_extent>
231  {
232  TETL_ASSERT(!(offset > size()));
233  TETL_ASSERT(!(count != dynamic_extent && count > size() - offset));
234  auto const sz = count == dynamic_extent ? size() - offset : count;
235  return {data() + offset, static_cast<size_type>(sz)};
236  }
237 
238 private:
239  pointer _data = nullptr;
240  size_type _size = 0;
241 };
