35template <typename R, typename... Args>
36struct inplace_func_vtable {
37 using storage_ptr_t = void*;
38
39 using invoke_ptr_t = R (*)(storage_ptr_t, Args&&...);
40 using process_ptr_t = void (*)(storage_ptr_t, storage_ptr_t);
41 using destructor_ptr_t = void (*)(storage_ptr_t);
42
43 invoke_ptr_t const invoke_ptr;
44 process_ptr_t const copy_ptr;
45 process_ptr_t const relocate_ptr;
46 destructor_ptr_t const destructor_ptr;
47
48 explicit constexpr inplace_func_vtable()
49 : invoke_ptr{[](storage_ptr_t /*p*/, Args&&... /*args*/) -> R {
50 etl::raise<etl::bad_function_call>("empty inplace_func_vtable");
51 }}
52 , copy_ptr{[](storage_ptr_t /*p*/, storage_ptr_t /*p*/) -> void {}}
53 , relocate_ptr{[](storage_ptr_t /*p*/, storage_ptr_t /*p*/) -> void {}}
54 , destructor_ptr{[](storage_ptr_t /*p*/) -> void {}}
55 {
56 }
57
58 template <typename C>
59 explicit constexpr inplace_func_vtable(wrapper<C> /*ignore*/)
60 : invoke_ptr{[](storage_ptr_t storagePtr, Args&&... args) -> R {
61 return (*static_cast<C*>(storagePtr))(static_cast<Args&&>(args)...);
62 }}
63 , copy_ptr{[](storage_ptr_t dstPtr, storage_ptr_t srcPtr) -> void {
64 ::new (dstPtr) C{(*static_cast<C*>(srcPtr))};
65 }}
66 , relocate_ptr{[](storage_ptr_t dstPtr, storage_ptr_t srcPtr) -> void {
67 ::new (dstPtr) C{etl::move(*static_cast<C*>(srcPtr))};
68 static_cast<C*>(srcPtr)->~C();
69 }}
70 , destructor_ptr{[](storage_ptr_t srcPtr) -> void { static_cast<C*>(srcPtr)->~C(); }}
71 {
72 }
73
74 inplace_func_vtable(inplace_func_vtable const&) = delete;
75 inplace_func_vtable(inplace_func_vtable&&) = delete;
76
77 auto operator=(inplace_func_vtable const&) -> inplace_func_vtable& = delete;
78 auto operator=(inplace_func_vtable&&) -> inplace_func_vtable& = delete;
79
80 ~inplace_func_vtable() = default;
81};

105template <typename R, typename... Args, size_t Capacity, size_t Alignment>
106struct inplace_function<R(Args...), Capacity, Alignment> {
107private:
108 using storage_t = aligned_storage_t<Capacity, Alignment>;
109 using vtable_t = detail::inplace_func_vtable<R, Args...>;
110 using vtable_ptr_t = vtable_t const*;
111
112 template <typename, size_t, size_t>
113 friend struct inplace_function;
114
115public:
116 using capacity = integral_constant<size_t, Capacity>;
117 using alignment = integral_constant<size_t, Alignment>;
118
120 inplace_function() noexcept : _vtable{addressof(detail::empty_vtable<R, Args...>)} { }
121
122 template <typename T, typename C = decay_t<T>>
123 requires(!detail::is_inplace_function<C>::value && is_invocable_r_v<R, C&, Args...>)
124 inplace_function(T&& closure)
125 {
126 static_assert(is_copy_constructible_v<C>, "inplace_function cannot be constructed from non-copyable type");
127 static_assert(
128 sizeof(C) <= Capacity,
129 "inplace_function cannot be constructed from object with this (large) size"
130 );
131 static_assert(
132 Alignment % alignof(C) == 0,
133 "inplace_function cannot be constructed from object with this (large) alignment"
134 );
135
136 static constexpr vtable_t const vt{detail::wrapper<C>{}};
137 _vtable = addressof(vt);
138
139 ::new (addressof(_storage)) C{etl::forward<T>(closure)};
140 }
141
142 template <size_t Cap, size_t Align>
143 inplace_function(inplace_function<R(Args...), Cap, Align> const& other)
144 : inplace_function{other._vtable, other._vtable->copy_ptr, addressof(other._storage)}
145 {
146 static_assert(
147 detail::is_valid_inplace_destination<Capacity, Alignment, Cap, Align>::value,
148 "conversion not allowed"
149 );
150 }
151
152 template <size_t Cap, size_t Align>
153 inplace_function(inplace_function<R(Args...), Cap, Align>&& other) noexcept
154 : inplace_function{other._vtable, other._vtable->relocate_ptr, addressof(other._storage)}
155 {
156 static_assert(
157 detail::is_valid_inplace_destination<Capacity, Alignment, Cap, Align>::value,
158 "conversion not allowed"
159 );
160 other._vtable = addressof(detail::empty_vtable<R, Args...>);
161 }
162
164 inplace_function(nullptr_t /*ignore*/) noexcept : _vtable{addressof(detail::empty_vtable<R, Args...>)} { }
165
166 inplace_function(inplace_function const& other) : _vtable{other._vtable}
167 {
168 _vtable->copy_ptr(addressof(_storage), addressof(other._storage));
169 }
170
171 inplace_function(inplace_function&& other) noexcept
172 : _vtable{exchange(other._vtable, addressof(detail::empty_vtable<R, Args...>))}
173 {
174 _vtable->relocate_ptr(addressof(_storage), addressof(other._storage));
175 }
176
179 auto operator=(nullptr_t) noexcept -> inplace_function&
180 {
181 _vtable->destructor_ptr(addressof(_storage));
182 _vtable = addressof(detail::empty_vtable<R, Args...>);
183 return *this;
184 }
185
186 auto operator=(inplace_function other) noexcept -> inplace_function&
187 {
188 _vtable->destructor_ptr(addressof(_storage));
189 _vtable = exchange(other._vtable, addressof(detail::empty_vtable<R, Args...>));
190 _vtable->relocate_ptr(addressof(_storage), addressof(other._storage));
191 return *this;
192 }
193
196 ~inplace_function() { _vtable->destructor_ptr(addressof(_storage)); }
197
199 auto operator()(Args... args) const -> R
200 {
201 return _vtable->invoke_ptr(addressof(_storage), etl::forward<Args>(args)...);
202 }
203
205 [[nodiscard]] explicit constexpr operator bool() const noexcept
206 {
207 return _vtable != addressof(detail::empty_vtable<R, Args...>);
208 }
209
211 auto swap(inplace_function& other) noexcept -> void
212 {
213 if (this == addressof(other)) {
214 return;
215 }
216
217 auto tmp = storage_t{};
218 _vtable->relocate_ptr(addressof(tmp), addressof(_storage));
219 other._vtable->relocate_ptr(addressof(_storage), addressof(other._storage));
220 _vtable->relocate_ptr(addressof(other._storage), addressof(tmp));
221
222 etl::swap(_vtable, other._vtable);
223 }
224
225private:
226 inplace_function(
227 vtable_ptr_t vtable,
228 typename vtable_t::process_ptr_t process,
229 typename vtable_t::storage_ptr_t storage
230 )
231 : _vtable{vtable}
232 {
233 process(addressof(_storage), storage);
234 }
235
236 vtable_ptr_t _vtable;
237 storage_t mutable _storage;
238};
