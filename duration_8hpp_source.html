215struct common_type<chrono::duration<Rep1, Period1>, chrono::duration<Rep2, Period2>> {

235template <typename Rep1, typename Period1, typename Rep2, typename Period2>
236[[nodiscard]] constexpr auto operator+(duration<Rep1, Period1> const& lhs, duration<Rep2, Period2> const& rhs)
237 -> common_type_t<duration<Rep1, Period1>, duration<Rep2, Period2>>
238{
239 using CD = common_type_t<duration<Rep1, Period1>, duration<Rep2, Period2>>;
