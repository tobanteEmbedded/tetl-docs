18 /// \brief Class template etl::chrono::duration represents a time interval.
19 ///
20 /// \details It consists of a count of ticks of type Rep and a tick period,
21 /// where the tick period is a compile-time rational constant representing the
22 /// number of seconds from one tick to the next. The only data stored in a
23 /// duration is a tick count of type Rep. If Rep is floating point, then the
24 /// duration can represent fractions of ticks. Period is included as part of the
25 /// duration's type, and is only used when converting between different
26 /// durations.
27 template <typename Rep, typename Period = etl::ratio<1>>
28 struct duration {
29  /// \brief Rep, an arithmetic type representing the number of ticks.
30  using rep = Rep;
31 
32  /// \brief A etl::ratio representing the tick period (i.e. the number of
33  /// seconds per tick).
34  using period = typename Period::type;
35 
36  /// \brief Constructs a new duration from one of several optional data
37  /// sources. The default constructor is defaulted.
38  constexpr duration() noexcept = default;
39 
40  /// \brief Constructs a new duration from one of several optional data
41  /// sources. The copy constructor is defaulted (makes a bitwise copy of the
42  /// tick count).
43  constexpr duration(duration const&) noexcept = default;
44 
45  /// \brief Constructs a duration with r ticks.
46  ///
47  /// \details Note that this constructor only participates in overload
48  /// resolution if const Rep2& (the argument type) is implicitly convertible
49  /// to rep (the type of this duration's ticks) and
50  /// treat_as_floating_point<rep>::value is true, or
51  /// treat_as_floating_point<Rep2>::value is false.
52  ///
53  /// That is, a duration with an integer tick count cannot be constructed
54  /// from a floating-point value, but a duration with a floating-point tick
55  /// count can be constructed from an integer value
56  template <typename Rep2>
57  requires(is_convertible_v<Rep2, rep> and (treat_as_floating_point_v<rep> or !treat_as_floating_point_v<Rep2>))
58  constexpr explicit duration(Rep2 const& r) noexcept : _rep(r)
59  {
60  }
61 
62  /// \brief Constructs a duration by converting d to an appropriate period
63  /// and tick count, as if by duration_cast<duration>(d).count().
64  ///
65  /// \details In order to prevent truncation during conversion, this
66  /// constructor only participates in overload resolution if computation of
67  /// the conversion factor (by etl::ratio_divide<Period2, Period>) does not
68  /// overflow and:
69  ///
70  /// treat_as_floating_point<rep>::value == true
71  ///
72  /// or both:
73  ///
74  /// ratio_divide<Period2, period>::den == 1, and
75  /// treat_as_floating_point<Rep2>::value == false
76  ///
77  /// That is, either the duration uses floating-point ticks, or Period2 is
78  /// exactly divisible by period
79  template <typename Rep2, typename Period2>
80  requires(
81  treat_as_floating_point_v<rep>
82  or (ratio_divide<Period2, period>::den == 1 and not treat_as_floating_point_v<Rep2>)
83  )
84  constexpr duration(duration<Rep2, Period2> const& other) noexcept
85  : _rep(static_cast<Rep>(other.count() * ratio_divide<Period2, period>::num))
86  {
87  }
88 
89  /// \brief Assigns the contents of one duration to another.
90  auto operator=(duration const& other) -> duration& = default;
91 
92  /// \brief Returns the number of ticks for this duration.
93  [[nodiscard]] constexpr auto count() const -> rep { return _rep; }
94 
95  /// \brief Returns a zero-length duration.
96  [[nodiscard]] static constexpr auto zero() noexcept -> duration
97  {
98  return duration(etl::chrono::duration_values<rep>::zero());
99  }
100 
101  /// \brief Returns a duration with the lowest possible value.
102  [[nodiscard]] static constexpr auto min() noexcept -> duration
103  {
104  return duration(etl::chrono::duration_values<rep>::min());
105  }
106 
107  /// \brief Returns a duration with the largest possible value.
108  [[nodiscard]] static constexpr auto max() noexcept -> duration
109  {
110  return duration(etl::chrono::duration_values<rep>::max());
111  }
112 
113  /// \brief Implements unary plus and unary minus for the durations.
114  [[nodiscard]] constexpr auto operator+() const -> etl::common_type_t<duration>
115  {
116  return etl::common_type_t<duration>(*this);
117  }
118 
119  /// \brief Implements unary plus and unary minus for the durations.
120  [[nodiscard]] constexpr auto operator-() const -> etl::common_type_t<duration>
121  {
122  return etl::common_type_t<duration>(-_rep);
123  }
124 
125  /// \brief Increments or decrements the number of ticks for this duration.
126  /// Equivalent to ++_rep; return *this;
127  constexpr auto operator++() -> duration&
128  {
129  ++_rep;
130  return *this;
131  }
132 
133  /// \brief Increments or decrements the number of ticks for this duration.
134  /// Equivalent to return duration(_rep++)
135  constexpr auto operator++(int) -> duration { return duration(_rep++); }
136 
137  /// \brief Increments or decrements the number of ticks for this duration.
138  /// Equivalent to --_rep; return *this;
139  constexpr auto operator--() -> duration&
140  {
141  --_rep;
142  return *this;
143  }
144 
145  /// \brief Increments or decrements the number of ticks for this duration.
146  /// Equivalent to return duration(_rep--);
147  constexpr auto operator--(int) -> duration { return duration(_rep--); }
148 
149  /// \brief Performs compound assignments between two durations with the same
150  /// period or between a duration and a tick count value.
151  constexpr auto operator+=(duration const& d) noexcept -> duration&
152  {
153  _rep += d.count();
154  return *this;
155  }
156 
157  /// \brief Performs compound assignments between two durations with the same
158  /// period or between a duration and a tick count value.
159  constexpr auto operator-=(duration const& d) noexcept -> duration&
160  {
161  _rep -= d.count();
162  return *this;
163  }
164 
165  /// \brief Performs compound assignments between two durations with the same
166  /// period or between a duration and a tick count value.
167  constexpr auto operator*=(rep const& rhs) noexcept -> duration&
168  {
169  _rep *= rhs;
170  return *this;
171  }
172 
173  /// \brief Performs compound assignments between two durations with the same
174  /// period or between a duration and a tick count value.
175  constexpr auto operator/=(rep const& rhs) noexcept -> duration&
176  {
177  _rep /= rhs;
178  return *this;
179  }
180 
181  /// \brief Performs compound assignments between two durations with the same
182  /// period or between a duration and a tick count value.
183  constexpr auto operator%=(rep const& rhs) noexcept -> duration&
184  {
185  _rep %= rhs;
186  return *this;
187  }
188 
189  /// \brief Performs compound assignments between two durations with the same
190  /// period or between a duration and a tick count value.
191  constexpr auto operator%=(duration const& rhs) noexcept -> duration&
192  {
193  _rep %= rhs.count();
194  return *this;
195  }
196 
197 private:
198  rep _rep{};
199 };

205 /// \brief Exposes the type named type, which is the common type of two
206 /// etl::chrono::durations, whose period is the greatest common divisor of
207 /// Period1 and Period2.
208 /// \details The period of the resulting duration can be computed by forming a
209 /// ratio of the greatest common divisor of Period1::num and Period2::num and
210 /// the least common multiple of Period1::den and Period2::den.
211 template <typename Rep1, typename Period1, typename Rep2, typename Period2>
212 struct common_type<chrono::duration<Rep1, Period1>, chrono::duration<Rep2, Period2>> {
213 private:
214  static constexpr auto num = gcd(Period1::num, Period2::num);
215  static constexpr auto den = lcm(Period1::den, Period2::den);
216 
217 public:
218  using type = chrono::duration<common_type_t<Rep1, Rep2>, ratio<num, den>>;
219 };

225 /// \brief Performs basic arithmetic operations between two durations or between
226 /// a duration and a tick count.
227 ///
228 /// \details Converts the two durations to their common type and creates a
229 /// duration whose tick count is the sum of the tick counts after conversion.
230 ///
231 /// https://en.cppreference.com/w/cpp/chrono/duration/operator_arith4
232 template <typename Rep1, typename Period1, typename Rep2, typename Period2>
233 [[nodiscard]] constexpr auto operator+(duration<Rep1, Period1> const& lhs, duration<Rep2, Period2> const& rhs)
234  -> common_type_t<duration<Rep1, Period1>, duration<Rep2, Period2>>
235 {
236  using CD = common_type_t<duration<Rep1, Period1>, duration<Rep2, Period2>>;
237  using CR = typename CD::rep;
238  return CD(static_cast<CR>(CD(lhs).count() + CD(rhs).count()));
239 }

241 /// \brief Performs basic arithmetic operations between two durations or between
242 /// a duration and a tick count.
243 ///
244 /// \details Converts the two durations to their common type and creates a
245 /// duration whose tick count is the rhs number of ticks subtracted from the lhs
246 /// number of ticks after conversion.
247 ///
248 /// https://en.cppreference.com/w/cpp/chrono/duration/operator_arith4
249 template <typename Rep1, typename Period1, typename Rep2, typename Period2>
250 [[nodiscard]] constexpr auto operator-(duration<Rep1, Period1> const& lhs, duration<Rep2, Period2> const& rhs)
251  -> common_type_t<duration<Rep1, Period1>, duration<Rep2, Period2>>
252 {
253  using CD = common_type_t<duration<Rep1, Period1>, duration<Rep2, Period2>>;
254  using CR = typename CD::rep;
255  return CD(static_cast<CR>(CD(lhs).count() - CD(rhs).count()));
256 }

258 /// \brief Performs basic arithmetic operations between two durations or between
259 /// a duration and a tick count.
260 ///
261 /// \details Converts the two durations to their common type and creates a
262 /// duration whose tick count is the rhs number of ticks subtracted from the lhs
263 /// number of ticks after conversion.
264 ///
265 /// https://en.cppreference.com/w/cpp/chrono/duration/operator_arith4
266 template <typename Rep1, typename Period1, typename Rep2, typename Period2>

268 operator/(duration<Rep1, Period1> const& lhs, duration<Rep2, Period2> const& rhs) -> common_type_t<Rep1, Rep2>
269 {

291 /// \brief Compares two durations. Checks if lhs and rhs are equal, i.e. the
292 /// number of ticks for the type common to both durations are equal.
293 template <typename Rep1, typename Period1, typename Rep2, typename Period2>
294 [[nodiscard]] constexpr auto operator==(duration<Rep1, Period1> const& lhs, duration<Rep2, Period2> const& rhs) -> bool
295 {
296  using common_t = typename etl::common_type<duration<Rep1, Period1>, duration<Rep2, Period2>>::type;
297  return common_t(lhs).count() == common_t(rhs).count();
298 }
