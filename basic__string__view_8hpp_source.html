315 [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
316 {
317 if (v.size() > size() - pos) {
318 return npos;
319 }
320
321 for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
322 if (unsafe_at(outerIdx) == v.front()) {
323 auto found = [&] {
324 for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
325 auto offset = outerIdx + innerIdx;
326 if (unsafe_at(offset) != v[innerIdx]) {
327 return false;
328 }
329 }
330
331 return true;
332 }();
333
334 if (found) {
335 return outerIdx;
336 }
337 }
338 }
339
340 return npos;
341 }

375 [[nodiscard]] constexpr auto rfind(basic_string_view sv, size_type pos = npos) const noexcept -> size_type
376 {
377 pos = etl::min(pos, size());
378 if (sv.size() < size() - pos) {
379 pos += sv.size();
380 } else {
381 pos = size();
382 }
383
384 auto const* r = etl::find_end(data(), data() + pos, sv.begin(), sv.end(), Traits::eq);
385 if (sv.size() > 0 && r == data() + pos) {
386 return npos;
387 }
388 return static_cast<size_type>(r - data());

552 [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
553 {
554 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
555 do { // NOLINT(cppcoreguidelines-avoid-do-while)
556 auto const current = unsafe_at(offset);
557 for (auto const ch : v) {
558 if (ch == current) {
559 return offset;
560 }
561 }
562 } while (offset-- != 0);
563
564 return npos;
565 }
