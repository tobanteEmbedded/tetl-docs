332 [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
333 {
334 if (v.size() > size() - pos) {
335 return npos;
336 }
337
338 for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
339 if (unsafe_at(outerIdx) == v.front()) {
340 auto found = [&] {
341 for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
342 auto offset = outerIdx + innerIdx;
343 if (unsafe_at(offset) != v[innerIdx]) {
344 return false;
345 }
346 }
347
348 return true;
349 }();
350
351 if (found) {
352 return outerIdx;
353 }
354 }
355 }
356
357 return npos;
358 }

548 [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
549 {
550 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
551 do { // NOLINT(cppcoreguidelines-avoid-do-while)
552 auto const current = unsafe_at(offset);
553 for (auto const ch : v) {
554 if (ch == current) {
555 return offset;
556 }
557 }
558 } while (offset-- != 0);
559
560 return npos;
561 }

607 [[nodiscard]] constexpr auto find_last_not_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
608 {
609 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
610 do { // NOLINT(cppcoreguidelines-avoid-do-while)
611 auto equals = [&](auto ch) { return ch == unsafe_at(offset); };
612 if (etl::none_of(v.begin(), v.end(), equals)) {
613 return offset;
614 }
615 } while (offset-- != 0);
616
617 return npos;
618 }
