341 [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
342 {
343 if (v.size() > size() - pos) {
344 return npos;
345 }
346
347 for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
348 if (unsafe_at(outerIdx) == v.front()) {
349 auto found = [&] {
350 for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
351 auto offset = outerIdx + innerIdx;
352 if (unsafe_at(offset) != v[innerIdx]) {
353 return false;
354 }
355 }
356
357 return true;
358 }();
359
360 if (found) {
361 return outerIdx;
362 }
363 }
364 }
365
366 return npos;
367 }

557 [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
558 {
559 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
560 do { // NOLINT(cppcoreguidelines-avoid-do-while)
561 auto const current = unsafe_at(offset);
562 for (auto const ch : v) {
563 if (ch == current) {
564 return offset;
565 }
566 }
567 } while (offset-- != 0);
568
569 return npos;
570 }

617 {

619 do { // NOLINT(cppcoreguidelines-avoid-do-while)
620 auto equals = [&](auto ch) { return ch == unsafe_at(offset); };
