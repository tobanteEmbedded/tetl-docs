340 [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
341 {
342 if (v.size() > size() - pos) {
343 return npos;
344 }
345
346 for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
347 if (unsafe_at(outerIdx) == v.front()) {
348 auto found = [&] {
349 for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
350 auto offset = outerIdx + innerIdx;
351 if (unsafe_at(offset) != v[innerIdx]) {
352 return false;
353 }
354 }
355
356 return true;
357 }();
358
359 if (found) {
360 return outerIdx;
361 }
362 }
363 }
364
365 return npos;
366 }

403 [[nodiscard]] constexpr auto rfind(basic_string_view sv, size_type pos = npos) const noexcept -> size_type
404 {
405 pos = etl::min(pos, size());
406 if (sv.size() < size() - pos) {
407 pos += sv.size();
408 } else {
409 pos = size();
410 }
411
412 auto const* r = etl::find_end(data(), data() + pos, sv.begin(), sv.end(), Traits::eq);
413 if (sv.size() > 0 && r == data() + pos) {
414 return npos;
415 }
416 return static_cast<size_type>(r - data());
417 }

424 [[nodiscard]] constexpr auto rfind(CharType c, size_type pos = npos) const noexcept -> size_type
425 {

431 ++pos;

580 [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
581 {
582 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
583 do { // NOLINT(cppcoreguidelines-avoid-do-while)
584 auto const current = unsafe_at(offset);
585 for (auto const ch : v) {
586 if (ch == current) {
587 return offset;
588 }
589 }
590 } while (offset-- != 0);
591
592 return npos;
593 }
