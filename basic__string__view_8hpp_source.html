324 [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
325 {
326 if (v.size() > size() - pos) {
327 return npos;
328 }
329
330 for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
331 if (unsafe_at(outerIdx) == v.front()) {
332 auto found = [&] {
333 for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
334 auto offset = outerIdx + innerIdx;
335 if (unsafe_at(offset) != v[innerIdx]) {
336 return false;
337 }
338 }
339
340 return true;
341 }();
342
343 if (found) {
344 return outerIdx;
345 }
346 }
347 }
348
349 return npos;
350 }

384 [[nodiscard]] constexpr auto rfind(basic_string_view sv, size_type pos = npos) const noexcept -> size_type
385 {
386 pos = etl::min(pos, size());
387 if (sv.size() < size() - pos) {
388 pos += sv.size();
389 } else {
390 pos = size();
391 }
392
393 auto const* r = etl::find_end(data(), data() + pos, sv.begin(), sv.end(), Traits::eq);
394 if (sv.size() > 0 && r == data() + pos) {
395 return npos;
396 }
397 return static_cast<size_type>(r - data());
398 }

561 [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
562 {
563 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
564 do { // NOLINT(cppcoreguidelines-avoid-do-while)
565 auto const current = unsafe_at(offset);
566 for (auto const ch : v) {
567 if (ch == current) {
568 return offset;
569 }
570 }
571 } while (offset-- != 0);
572
573 return npos;
574 }
