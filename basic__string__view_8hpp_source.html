235 {

336 [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
337 {
338 if (v.size() > size() - pos) {
339 return npos;
340 }
341
342 for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
343 if (unsafe_at(outerIdx) == v.front()) {
344 auto found = [&] {
345 for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
346 auto offset = outerIdx + innerIdx;
347 if (unsafe_at(offset) != v[innerIdx]) {
348 return false;
349 }
350 }
351
352 return true;
353 }();
354
355 if (found) {
356 return outerIdx;
357 }
358 }
359 }
360
361 return npos;
362 }

396 [[nodiscard]] constexpr auto rfind(basic_string_view sv, size_type pos = npos) const noexcept -> size_type
397 {
398 pos = etl::min(pos, size());
399 if (sv.size() < size() - pos) {
400 pos += sv.size();
401 } else {
402 pos = size();
403 }
404
405 auto const* r = etl::find_end(data(), data() + pos, sv.begin(), sv.end(), Traits::eq);
406 if (sv.size() > 0 && r == data() + pos) {
407 return npos;
408 }
409 return static_cast<size_type>(r - data());
410 }

573 [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
574 {
575 auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
576 do { // NOLINT(cppcoreguidelines-avoid-do-while)
577 auto const current = unsafe_at(offset);
578 for (auto const ch : v) {
579 if (ch == current) {
580 return offset;
581 }
582 }
583 } while (offset-- != 0);
584
585 return npos;
586 }
