25 /// \brief The class template basic_string_view describes an object that can
26 /// refer to a constant contiguous sequence of char-like objects with the first
27 /// element of the sequence at position zero. A typical implementation holds

214  /// \brief Compares two character sequences.
215  ///
216  /// https://en.cppreference.com/w/cpp/string/basic_string_view/compare
217  [[nodiscard]] constexpr auto compare(basic_string_view v) const noexcept -> int
218  {
219  auto const rlen = etl::min(size(), v.size());
220  auto const res = traits_type::compare(data(), v.data(), rlen);
221 
222  if (res < 0) {
223  return -1;
224  }
225  if (res > 0) {
226  return 1;
227  }
228 
229  if (size() < v.size()) {
230  return -1;
231  }
232  if (size() > v.size()) {
233  return 1;
234  }
235 
236  return 0;
237  }

325  /// \brief Finds the first substring equal to the given character sequence.
326  /// Finds the first occurence of v in this view, starting at position pos.
327  ///
328  /// \returns Position of the first character of the found substring, or npos
329  /// if no such substring is found.
330  [[nodiscard]] constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept -> size_type
331  {
332  if (v.size() > size() - pos) {
333  return npos;
334  }
335 
336  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx) {
337  if (unsafe_at(outerIdx) == v.front()) {
338  auto found = [&] {
339  for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx) {
340  auto offset = outerIdx + innerIdx;
341  if (unsafe_at(offset) != v[innerIdx]) {
342  return false;
343  }
344  }
345 
346  return true;
347  }();
348 
349  if (found) {
350  return outerIdx;
351  }
352  }
353  }
354 
355  return npos;
356  }

538  /// \brief Finds the last character equal to one of characters in the given
539  /// character sequence. Exact search algorithm is not specified. The search
540  /// considers only the interval [0; pos]. If the character is not present in
541  /// the interval, npos will be returned. Finds the last occurence of any of
542  /// the characters of v in this view, ending at position pos.
543  ///

545  /// substring, or npos if no such character is found.
546  [[nodiscard]] constexpr auto find_last_of(basic_string_view v, size_type pos = npos) const noexcept -> size_type
547  {
548  auto offset = etl::clamp<size_type>(pos, 0, size() - 1);
549  do { // NOLINT(cppcoreguidelines-avoid-do-while)
550  auto const current = unsafe_at(offset);
551  for (auto const ch : v) {
552  if (ch == current) {
553  return offset;
554  }
555  }
556  } while (offset-- != 0);
557 
558  return npos;
559  }
