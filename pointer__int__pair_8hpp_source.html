13 /// \brief This struct implements a pair of a pointer and small integer. It is
14 /// designed to represent this in the space required by one pointer by
15 /// bitmangling the integer into the low part of the pointer. This can only be
16 /// done for small integers: typically up to 3 bits, but it depends on the
17 /// number of bits available according to pointer_like_traits for the type.
18 ///
19 /// \details Note that pointer_int_pair always puts the IntVal part in the
20 /// highest bits possible. For example, pointer_int_pair<void*, 1, bool> will
21 /// put the bit for the bool into bit #2, not bit #0, which allows the low two
22 /// bits to be used for something else. For example, this allows:
23 /// pointer_int_pair<pointer_int_pair<void*, 1, bool>, 1, bool>
24 /// ... and the two bools will land in different bits.
25 ///
26 /// \include memory.cpp
27 template <
28  typename PointerT,
29  unsigned IntBits,
30  typename IntType = unsigned,
31  typename PtrTraits = pointer_like_traits<PointerT>,
32  typename Info = pointer_int_pair_info<PointerT, IntBits, PtrTraits>>
33 struct pointer_int_pair {
34  using pointer_type = PointerT;
35  using pointer_traits = PtrTraits;
36  using pointer_info = Info;
37  using int_type = IntType;
38  static constexpr auto int_bits = IntBits;
39 
40  constexpr pointer_int_pair() = default;
