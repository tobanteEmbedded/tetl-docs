40template <typename Int, to_integer_options Options = to_integer_options{}>
41[[nodiscard]] constexpr auto to_integer(etl::string_view str, Int base = Int(10)) noexcept -> to_integer_result<Int>
42{
43 auto len = str.size();
44
45 auto i = size_t{};
46 if constexpr (Options.skip_whitespace) {
47 while ((len != 0) and etl::isspace(static_cast<int>(str[i])) and (str[i] != char(0))) {
48 ++i;
49 --len;
50 }
51 }
52
53 if (len == 0 or str[i] == char(0)) {
54 return {.end = str.data(), .error = to_integer_error::invalid_input, .value = Int{}};
55 }
56
57 // optional minus for signed types
58 [[maybe_unused]] auto sign = Int(1);
59 if constexpr (is_signed_v<Int>) {
60 if (((len != 0) and (str[i] != char(0))) and (str[i] == '-')) {
61 sign = Int(-1);
62 ++i;
63 --len;
64 }
65 }
66
67 auto const firstDigit = i;
68
69 // loop over digits
70 auto value = Int{};
71 for (; (str[i] != char(0)) and (len != 0); ++i, --len) {
72
73 auto digit = Int{};
74 if (etl::isdigit(static_cast<int>(str[i]))) {
75 digit = static_cast<Int>(str[i] - '0');
76 } else if (etl::isalpha(static_cast<int>(str[i]))) {
77 auto const x = static_cast<char>(etl::tolower(static_cast<int>(str[i])));
78 digit = static_cast<Int>(static_cast<Int>(x) - static_cast<Int>('a') + 10);
79 } else {
80 break;
81 }
82
83 if (digit >= base) {
84 if (i != firstDigit) {
85 break;
86 }
87 return {.end = str.data(), .error = to_integer_error::invalid_input, .value = Int{}};
88 }
89
90 // TODO(tobi): Check overflow
91 value = static_cast<Int>(value * base + digit);
92 }
93
94 if constexpr (is_signed_v<Int>) {
95 value *= sign;
96 }
97
98 return {.end = etl::addressof(str[i]), .error = to_integer_error::none, .value = value};
99}
