197// One deduction guide is provided for array to provide an equivalent of
198// experimental::make_array for construction of array from a variadic parameter
199// pack. The program is ill-formed if (is_same_v<T, U> and ...) is not true.
200// Note that it is true when sizeof...(U) is zero.
201template <typename T, typename... U>
202array(T, U...) -> array<T, 1 + sizeof...(U)>;
