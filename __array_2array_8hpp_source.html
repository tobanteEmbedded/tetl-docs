24 /// \brief array is a container that encapsulates fixed size arrays.
25 ///
26 /// \details This container is an aggregate type with the same semantics as a
27 /// struct holding a C-style array Type[N] as its only non-static data member.
28 /// Unlike a C-style array, it doesn't decay to Type* automatically. As an
29 /// aggregate type, it can be initialized with aggregate-initialization given at
30 /// most N initializers that are convertible to
31 /// Type: `array<int, 3> a = {1,2,3};`
32 ///
33 /// \include array.cpp
34 template <typename Type, size_t Size>

181  /// \brief Exchanges the contents of the container with those of other. Does
182  /// not cause iterators and references to associate with the other
183  /// container.
184  constexpr auto swap(array& other) noexcept(is_nothrow_swappable_v<Type>) -> void
185  {
186  using etl::swap;
187  for (auto i = size_type{0}; i < size(); ++i) {
188  swap((*this)[i], other[i]);

190  }
