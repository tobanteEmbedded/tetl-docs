16 /// \brief Returns half the sum of a + b. If the sum is odd, the result is
17 /// rounded towards a.
18 ///
19 /// \details CppCon 2019: Marshall Clow "midpoint? How Hard Could it Be?"
20 /// Integer version was updated to match implementation from libc++.
21 ///
22 /// https://www.youtube.com/watch?v=sBtAGxBh-XI)
23 /// https://en.cppreference.com/w/cpp/numeric/midpoint
24 template <typename Int>
25  requires(etl::is_integral_v<Int> and not etl::is_same_v<Int, bool>)
26 constexpr auto midpoint(Int a, Int b) noexcept -> Int
27 {
28  using UInt = etl::make_unsigned_t<Int>;
29 
30  auto const shift = static_cast<UInt>(etl::numeric_limits<UInt>::digits - 1);
31  auto const diff = static_cast<UInt>(UInt(b) - UInt(a));
32  auto const sign = static_cast<UInt>(b < a);
33  auto const half = static_cast<UInt>((diff / 2) + (sign << shift) + (sign & diff));
34 
35  return a + static_cast<Int>(half);
36 }

38 template <etl::floating_point Float>
39 constexpr auto midpoint(Float a, Float b) noexcept -> Float
40 {
41  auto const lo = etl::numeric_limits<Float>::min() * 2;
42  auto const hi = etl::numeric_limits<Float>::max() / 2;
43 
44  if (etl::abs(a) <= hi and etl::abs(b) <= hi) {
45  return (a + b) / 2;
46  }
47  if (etl::abs(a) < lo) {
48  return a + b / 2;
49  }
50  if (etl::abs(b) < lo) {
51  return a / 2 + b;
52  }
53 
54  return a / 2 + b / 2;
55 }
