12 /// \brief etl::back_insert_iterator is a LegacyOutputIterator that appends to a
13 /// container for which it was constructed. The container's push_back() member
14 /// function is called whenever the iterator (whether dereferenced or not) is
15 /// assigned to. Incrementing the etl::back_insert_iterator is a no-op.
16 template <typename Container>
17 struct back_insert_iterator {
18  using iterator_category = output_iterator_tag;
19  using value_type = void;
20  using difference_type = ptrdiff_t;
21  using pointer = void;
22  using reference = void;
23  using container_type = Container;
24 
25  /// \brief Initializes the underlying pointer to container with nullptr.
26  constexpr back_insert_iterator() noexcept = default;
27 
28  /// \brief Initializes the underlying pointer to the container to
29  /// etl::addressof(c).
30  constexpr explicit back_insert_iterator(Container& container) : _container{etl::addressof(container)} { }
31 
32  /// \brief Inserts the given value value to the container.
33  constexpr auto operator=(typename Container::value_type const& value) -> back_insert_iterator&
34  {
35  _container->push_back(value);
36  return *this;
37  }
38 
39  /// \brief Inserts the given value value to the container.
40  constexpr auto operator=(typename Container::value_type&& value) -> back_insert_iterator&
41  {
42  _container->push_back(TETL_MOVE(value));
43  return *this;
44  }
45 
46  /// \brief Does nothing, this member function is provided to satisfy the
47  /// requirements of LegacyOutputIterator. It returns the iterator itself,
48  /// which makes it possible to use code such as *iter = value to output
49  /// (insert) the value into the underlying container.
50  constexpr auto operator*() -> back_insert_iterator& { return *this; }
51 
52  /// \brief Does nothing. These operator overloads are provided to satisfy
53  /// the requirements of LegacyOutputIterator. They make it possible for the
54  /// expressions *iter++=value and *++iter=value to be used to output
55  /// (insert) a value into the underlying container.
56  constexpr auto operator++() -> back_insert_iterator& { return *this; }
57 
58  /// \brief Does nothing. These operator overloads are provided to satisfy
59  /// the requirements of LegacyOutputIterator. They make it possible for the
60  /// expressions *iter++=value and *++iter=value to be used to output
61  /// (insert) a value into the underlying container.
62  constexpr auto operator++(int) -> back_insert_iterator { return *this; }
63 
64 private:

66 };
