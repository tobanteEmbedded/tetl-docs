73 constexpr basic_bitset(unsigned long long val) noexcept
74 {
75 auto const digits = static_cast<size_t>(etl::numeric_limits<unsigned long long>::digits);
76 auto const m = etl::min(digits, size());
77 for (auto i = etl::size_t(0); i < m; ++i) {
78 unchecked_set(i, etl::test_bit(val, static_cast<unsigned long long>(i)));
79 }
80 }

97 [[nodiscard]] constexpr auto all() const noexcept -> bool
98 {
99 if constexpr (not has_padding) {
100 return etl::all_of(_words.cbegin(), _words.cend(), [](auto word) { return word == ones; });
101 } else {
102 return count() == size();
103 }

107 [[nodiscard]] constexpr auto any() const noexcept -> bool { return not none(); }

110 [[nodiscard]] constexpr auto none() const noexcept -> bool

116 [[nodiscard]] constexpr auto count() const noexcept -> etl::size_t
117 {
118 return etl::transform_reduce(_words.cbegin(), _words.cend(), etl::size_t(0), etl::plus(), [](auto word) {
119 return static_cast<etl::size_t>(etl::popcount(word));
120 });
121 }

125 [[nodiscard]] constexpr auto unchecked_test(etl::size_t pos) const -> bool
126 {

131 constexpr auto set() noexcept -> basic_bitset&

147 constexpr auto unchecked_set(etl::size_t pos, bool value = true) -> basic_bitset&

238private:
239 static constexpr auto ones = etl::numeric_limits<WordType>::max();
240 static constexpr auto bits_per_word = static_cast<size_t>(etl::numeric_limits<WordType>::digits);
241 static constexpr auto num_words = (Bits + bits_per_word - 1) / bits_per_word;
242 static constexpr auto has_padding = (Bits % bits_per_word) != 0;
243
244 [[nodiscard]] static constexpr auto word_index(etl::size_t pos) -> etl::size_t { return pos / bits_per_word; }
245
246 [[nodiscard]] static constexpr auto offset_in_word(etl::size_t pos) -> WordType
247 {
248 return static_cast<WordType>(pos & (bits_per_word - etl::size_t(1)));
249 }
250
251 constexpr auto transform_bit(etl::size_t pos, auto op) -> basic_bitset&
252 {
253 auto& word = _words[word_index(pos)];
254 word = op(word, offset_in_word(pos));
255 return *this;
256 }
257
258 etl::array<WordType, num_words> _words{};
259};
