110/// The char_traits class is a traits class template that abstracts basic
111/// character and string operations for a given character type.
112///
113/// The defined operation set is such that generic algorithms almost always can be
114/// implemented in terms of it. It is thus possible to use such algorithms with
115/// almost any possible character or string type, just by supplying a customized
116/// char_traits class. The char_traits class template serves as a basis for
117/// explicit instantiations. The user can provide a specialization for any
118/// custom character types. Several specializations are defined for the standard
119/// character types. If an operation on traits emits an exception, the behavior
120/// is undefined.
121///
122/// \headerfile etl/string.hpp
123/// \ingroup string
124template <typename CharT>
125struct char_traits;
126
127/// Specializations of char_traits for type char.
128/// \ingroup string
129template <>
130struct char_traits<char> : detail::char_traits_base<char, int, -1> { };
131
132/// Specializations of char_traits for type wchar_t.
133/// \ingroup string
