159 constexpr auto insert(value_type&& value) -> pair<iterator, bool>
160 requires(is_move_constructible_v<value_type>)
161 {
162 if (!full()) {
163 auto cmp = key_compare{};
164 auto* p = etl::lower_bound(_memory.begin(), _memory.end(), value, cmp);
165 if (p == _memory.end() || *(p) != value) {
166 _memory.push_back(TETL_MOVE(value));
167 auto* pos = rotate(p, _memory.end() - 1, _memory.end());
168 return make_pair(pos, true);
169 }
170 }
171
172 return pair<iterator, bool>(nullptr, false);
173 }
