165 constexpr auto insert(value_type&& value) -> pair<iterator, bool>
166 requires(is_move_constructible_v<value_type>)
167 {
168 if (!full()) {
169 auto cmp = key_compare{};
170 auto* p = etl::lower_bound(_memory.begin(), _memory.end(), value, cmp);
171 if (p == _memory.end() || *(p) != value) {
172 _memory.push_back(etl::move(value));
173 auto* pos = rotate(p, _memory.end() - 1, _memory.end());
174 return make_pair(pos, true);
175 }
176 }
177
178 return pair<iterator, bool>(nullptr, false);
179 }
