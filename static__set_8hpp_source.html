161  constexpr auto insert(value_type&& value) -> enable_if_t<is_move_constructible_v<value_type>, pair<iterator, bool>>
162  {
163  if (!full()) {
164  auto cmp = key_compare {};
165  auto* p = etl::lower_bound(memory_.begin(), memory_.end(), value, cmp);
166  if (p == memory_.end() || *(p) != value) {
167  memory_.push_back(move(value));
168  auto* pos = rotate(p, memory_.end() - 1, memory_.end());
169  return make_pair(pos, true);
170  }
171  }
172 
173  return pair<iterator, bool>(nullptr, false);
174  }
