35template <typename Int, skip_whitespace Skip = skip_whitespace::yes>
36[[nodiscard]] constexpr auto
37string_to_integer(char const* str, size_t len, Int base = Int(10)) noexcept -> string_to_integer_result<Int>
38{
39 if (*str == char(0)) {
40 return {
41 .end = str,
42 .error = string_to_integer_error::invalid_input,
43 .value = Int{},
44 };
45 }
46
47 auto i = size_t{};
48 if constexpr (Skip == skip_whitespace::yes) {
49 while (isspace(static_cast<int>(str[i])) and (len != 0) and (str[i] != char(0))) {
50 ++i;
51 --len;
52 }
53 }
54
55 // optional minus for signed types
56 [[maybe_unused]] auto sign = Int(1);
57 if constexpr (is_signed_v<Int>) {
58 if (((len != 0) and (str[i] != char(0))) and (str[i] == '-')) {
59 sign = Int(-1);
60 ++i;
61 --len;
62 }
63 }
64
65 auto const firstDigit = i;
66
67 // loop over digits
68 auto value = Int{};
69 for (; (str[i] != char(0)) and (len != 0); ++i, --len) {
70
71 auto digit = Int{};
72 if (isdigit(static_cast<int>(str[i]))) {
73 digit = static_cast<Int>(str[i] - '0');
74 } else if (isalpha(static_cast<int>(str[i]))) {
75 auto const x = static_cast<char>(tolower(static_cast<int>(str[i])));
76 digit = static_cast<Int>(static_cast<Int>(x) - static_cast<Int>('a') + 10);
77 } else {
78 break;
79 }
80
81 if (digit >= base) {
82 if (i != firstDigit) {
83 break;
84 }
85 return {
86 .end = str,
87 .error = string_to_integer_error::invalid_input,
88 .value = Int{},
89 };
90 }
91
92 // TODO(tobi): Check overflow
93 value = static_cast<Int>(value * base + digit);
94 }
95
96 if constexpr (is_signed_v<Int>) {
97 value *= sign;
98 }
99
100 return {
101 .end = &str[i],
102 .error = string_to_integer_error::none,
103 .value = value,
104 };
105}

113template <typename FloatT>
114[[nodiscard]] constexpr auto string_to_floating_point(char const* str, char const** last = nullptr) noexcept -> FloatT
115{
116 auto res = FloatT{0};
117 auto div = FloatT{1};
118 auto afterDecimalPoint = false;
119 auto leadingSpaces = true;
120
121 auto const* ptr = str;
122 for (; *ptr != '\0'; ++ptr) {
123 if (etl::isspace(*ptr) && leadingSpaces) {
124 continue;
125 }
126 leadingSpaces = false;
127
128 if (etl::isdigit(*ptr)) {
129 if (!afterDecimalPoint) {
130 res *= 10; // Shift the previous digits to the left
131 res += *ptr - '0'; // Add the new one
132 } else {
133 div *= 10;
134 res += static_cast<FloatT>(*ptr - '0') / div;
135 }
136 } else if (*ptr == '.') {
137 afterDecimalPoint = true;
138 } else {
139 break;
140 }
141 }
142
143 if (last != nullptr) {
144 *last = ptr;
145 }
146
147 return res;
148}

160template <typename Int, bool TerminateWithNull = true>
161[[nodiscard]] constexpr auto
162integer_to_string(Int num, char* str, int base = 10, size_t length = etl::numeric_limits<size_t>::max())
163 -> integer_to_string_result
164{
165 auto reverseString = [](char* string, etl::size_t len) {
166 etl::size_t f = 0;
167 etl::size_t l = len - 1;
168 while (f < l) {
169 auto const tmp = string[f];
170 string[f] = string[l];
171 string[l] = tmp;
172 f++;
173 l--;
174 }
175 };
176
177 // Handle 0 explicitely, otherwise empty string is printed for 0
178 etl::size_t i = 0;
179 if (num == 0) {
180 if (length < (1 + static_cast<size_t>(TerminateWithNull))) {
181 return {str + length, integer_to_string_error::overflow};
182 }
183 str[i++] = '0';
184 if constexpr (TerminateWithNull) {
185 str[i] = '\0';
186 }
187 return {&str[i]};
188 }
189
190 bool isNegative = false;
191 if constexpr (etl::is_signed_v<Int>) {
192 if (num < 0 && base == 10) {
193 isNegative = true;
194 num = -num;
195 }
196 }
197
198 while (num != 0) {
199 auto const rem = static_cast<char>(num % static_cast<Int>(base));
200 str[i++] = (rem > 9) ? (rem - 10) + 'a' : rem + '0';
201 num = num / static_cast<Int>(base);
202
203 if (length <= i) {
204 return {nullptr, integer_to_string_error::overflow};
205 }
206 }
207
208 if constexpr (etl::is_signed_v<Int>) {
209 if (isNegative) {
210 str[i++] = '-';
211 }
212 }
213
214 if constexpr (TerminateWithNull) {
215 str[i] = '\0';
216 }
217
218 reverseString(str, i);
219 return {&str[i]};
220}
