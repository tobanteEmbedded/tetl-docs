36 template <typename Int, skip_whitespace Skip = skip_whitespace::yes>
37 [[nodiscard]] constexpr auto
38 string_to_integer(char const* str, size_t len, Int base = Int(10)) noexcept -> string_to_integer_result<Int>
39 {
40  if (*str == char(0)) {
41  return {
42  .end = str,
43  .error = string_to_integer_error::invalid_input,
44  .value = Int{},
45  };
46  }
47 
48  auto i = size_t{};
49  if constexpr (Skip == skip_whitespace::yes) {
50  while (isspace(static_cast<int>(str[i])) and (len != 0) and (str[i] != char(0))) {
51  ++i;
52  --len;
53  }
54  }
55 
56  // optional minus for signed types
57  [[maybe_unused]] auto sign = Int(1);
58  if constexpr (is_signed_v<Int>) {
59  if (((len != 0) and (str[i] != char(0))) and (str[i] == '-')) {
60  sign = Int(-1);
61  ++i;
62  --len;
63  }
64  }
65 
66  auto const firstDigit = i;
67 
68  // loop over digits
69  auto value = Int{};
70  for (; (str[i] != char(0)) and (len != 0); ++i, --len) {
71 
72  auto digit = Int{};
73  if (isdigit(static_cast<int>(str[i]))) {
74  digit = static_cast<Int>(str[i] - '0');
75  } else if (isalpha(static_cast<int>(str[i]))) {
76  auto const x = static_cast<char>(tolower(static_cast<int>(str[i])));
77  digit = static_cast<Int>(static_cast<Int>(x) - static_cast<Int>('a') + 10);
78  } else {
79  break;
80  }
81 
82  if (digit >= base) {
83  if (i != firstDigit) {
84  break;
85  }
86  return {
87  .end = str,
88  .error = string_to_integer_error::invalid_input,
89  .value = Int{},
90  };
91  }
92 
93  // TODO(tobi): Check overflow
94  value = static_cast<Int>(value * base + digit);
95  }
96 
97  if constexpr (is_signed_v<Int>) {
98  value *= sign;
99  }
100 
101  return {
102  .end = &str[i],
103  .error = string_to_integer_error::none,
104  .value = value,
105  };
106 }

114 template <typename FloatT>
115 [[nodiscard]] constexpr auto string_to_floating_point(char const* str, char const** last = nullptr) noexcept -> FloatT
116 {
117  auto res = FloatT{0};
118  auto div = FloatT{1};
119  auto afterDecimalPoint = false;
120  auto leadingSpaces = true;
121 
122  auto const* ptr = str;
123  for (; *ptr != '\0'; ++ptr) {
124  if (etl::isspace(*ptr) && leadingSpaces) {
125  continue;
126  }
127  leadingSpaces = false;
128 
129  if (etl::isdigit(*ptr)) {
130  if (!afterDecimalPoint) {
131  res *= 10; // Shift the previous digits to the left
132  res += *ptr - '0'; // Add the new one
133  } else {
134  div *= 10;
135  res += static_cast<FloatT>(*ptr - '0') / div;
136  }
137  } else if (*ptr == '.') {
138  afterDecimalPoint = true;
139  } else {
140  break;
141  }
142  }
143 
144  if (last != nullptr) {
145  *last = ptr;
146  }
147 
148  return res;
149 }

161 template <typename Int, bool TerminateWithNull = true>
162 [[nodiscard]] constexpr auto
163 integer_to_string(Int num, char* str, int base = 10, size_t length = etl::numeric_limits<size_t>::max())
164  -> integer_to_string_result
165 {
166  auto reverseString = [](char* string, etl::size_t len) {
167  etl::size_t f = 0;
168  etl::size_t l = len - 1;
169  while (f < l) {
170  auto const tmp = string[f];
171  string[f] = string[l];
172  string[l] = tmp;
173  f++;
174  l--;
175  }
176  };
177 
178  // Handle 0 explicitely, otherwise empty string is printed for 0
179  etl::size_t i = 0;
180  if (num == 0) {
181  if (length < (1 + static_cast<size_t>(TerminateWithNull))) {
182  return {str + length, integer_to_string_error::overflow};
183  }
184  str[i++] = '0';
185  if constexpr (TerminateWithNull) {
186  str[i] = '\0';
187  }
188  return {&str[i]};
189  }
190 
191  bool isNegative = false;
192  if constexpr (etl::is_signed_v<Int>) {
193  if (num < 0 && base == 10) {
194  isNegative = true;
195  num = -num;
196  }
197  }
198 
199  while (num != 0) {
200  auto const rem = static_cast<char>(num % static_cast<Int>(base));
201  str[i++] = (rem > 9) ? (rem - 10) + 'a' : rem + '0';
202  num = num / static_cast<Int>(base);
203 
204  if (length <= i) {
205  return {nullptr, integer_to_string_error::overflow};
206  }
207  }
208 
209  if constexpr (etl::is_signed_v<Int>) {
210  if (isNegative) {
211  str[i++] = '-';
212  }
213  }
214 
215  if constexpr (TerminateWithNull) {
216  str[i] = '\0';
217  }
218 
219  reverseString(str, i);
220  return {&str[i]};
221 }
