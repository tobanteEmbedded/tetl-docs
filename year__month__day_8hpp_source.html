18  // https://howardhinnant.github.io/date_algorithms.html#civil_from_days
19  template <typename Int>
20  [[nodiscard]] static constexpr auto civil_from_days(Int z) noexcept -> year_month_day
21  {
22  static_assert(etl::numeric_limits<uint32_t>::digits >= 18, "Not yet ported to a 16 bit unsigned integer");
23  static_assert(etl::numeric_limits<Int>::digits >= 20, "Not yet ported to a 16 bit signed integer");
24 
25  z += 719468;
26  Int const era = (z >= 0 ? z : z - 146096) / 146097;
27  auto const doe = static_cast<uint32_t>(z - era * 146097);
28  uint32_t const yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
29  Int const y = static_cast<Int>(yoe) + era * 400;
30  uint32_t const doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
31  uint32_t const mp = (5 * doy + 2) / 153;
32  uint32_t const d = doy - (153 * mp + 2) / 5 + 1;
33  uint32_t const m = mp < 10 ? mp + 3 : mp - 9;
34 
35  return {
36  chrono::year{y + (m <= 2)},
37  chrono::month{m},
38  chrono::day{d},
39  };
40  }

42  // https://howardhinnant.github.io/date_algorithms.html#days_from_civil
43  template <typename Int>
44  [[nodiscard]] static constexpr auto days_from_civil(Int y, uint32_t m, uint32_t d) noexcept -> days
45  {
46  static_assert(etl::numeric_limits<uint32_t>::digits >= 18, "Not yet ported to a 16 bit unsigned integer");
47  static_assert(etl::numeric_limits<Int>::digits >= 20, "Not yet ported to a 16 bit signed integer");
48  y -= m <= 2;
49  Int const era = (y >= 0 ? y : y - 399) / 400;
50  auto const yoe = static_cast<uint32_t>(y - era * 400); // [0, 399]
51  uint32_t const doy = (153 * (m > 2 ? m - 3 : m + 9) + 2) / 5 + d - 1; // [0, 365]
52  uint32_t const doe = yoe * 365 + yoe / 4 - yoe / 100 + doy; // [0, 146096]
53  return days{era * 146097 + static_cast<Int>(doe) - 719468};
54  }
