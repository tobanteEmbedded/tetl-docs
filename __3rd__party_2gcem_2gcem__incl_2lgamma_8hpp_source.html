82 template <typename T>
83 constexpr auto lgamma_check(T const x) noexcept -> T
84 {
85  return ( // NaN check
86  is_nan(x) ? etl::numeric_limits<T>::quiet_NaN() :
87  // indistinguishable from one or <= zero
88  etl::numeric_limits<T>::epsilon() > abs(x - T(1)) ? T(0)
89  : etl::numeric_limits<T>::epsilon() > x ? etl::numeric_limits<T>::infinity()
90  :
91  // else
92  lgamma_begin(x - T(1))
93  );
94 }

98 /**
99  * Compile-time log-gamma function
100  *
101  * @param x a real-valued input.
102  * @return computes the log-gamma function
103  * \f[ \ln \Gamma(x) = \ln \int_0^\infty y^{x-1} \exp(-y) dy \f]
104  * using a polynomial form:
105  * \f[ \Gamma(x+1) \approx (x+g+0.5)^{x+0.5} \exp(-x-g-0.5) \sqrt{2 \pi} \left[
106  * c_0 + \frac{c_1}{x+1} + \frac{c_2}{x+2} + \cdots + \frac{c_n}{x+n} \right]
107  * \f] where the value \f$ g \f$ and the coefficients \f$ (c_0, c_1, \ldots,
108  * c_n) \f$ are taken from Paul Godfrey, whose note can be found here:
109  * http://my.fit.edu/~gabdo/gamma.txt
110  */
111 
112 template <typename T>
113 constexpr auto lgamma(T const x) noexcept -> return_t<T>
114 {
115  return internal::lgamma_check(static_cast<return_t<T>>(x));
116 }
