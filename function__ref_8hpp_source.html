22 /// Non-owning view of a callable.
23 ///
24 /// http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2019/p0792r4.html
25 /// https://github.com/TartanLlama/function_ref
26 template <typename R, typename... Args>
27 struct function_ref<R(Args...)> {
28 private:
29  using internal_signature_t = R (*)(void*, Args...);
30 
31  void* _obj{nullptr};
32  internal_signature_t _callable{nullptr};
33 
34 public:
35  /// \brief Constructs a function_ref referring to f.
36  template <typename F>
37  requires(not is_same_v<decay_t<F>, function_ref> and is_invocable_r_v<R, F &&, Args...>)
38  function_ref(F&& f) noexcept
39  : _obj(const_cast<void*>(reinterpret_cast<void const*>(addressof(f))))
40  , _callable{
41  +[](void* obj, Args... args) -> R {
42  auto* func = reinterpret_cast<add_pointer_t<F>>(obj);
43  return invoke(*func, TETL_FORWARD(args)...);
44  },
45  }
46  {
47  }
48 
49  /// \brief Reassigns this function_ref to refer to f.
50  template <typename F>
51  requires(is_invocable_r_v<R, F &&, Args...>)
52  auto operator=(F&& f) noexcept -> function_ref&
53  {
54  _obj = reinterpret_cast<void*>(addressof(f));
55  _callable = +[](void* obj, Args... args) {
56  auto* func = reinterpret_cast<add_pointer_t<F>>(obj);
57  return invoke(*func, TETL_FORWARD(args)...);
58  };
59 
60  return *this;
61  }
62 
63  function_ref(function_ref const& /*other*/) = default;
64 
65  auto operator=(function_ref const& /*other*/) -> function_ref& = default;
66 
67  /// Exchanges the values of *this and rhs.
68  auto swap(function_ref& other) noexcept -> void
69  {
70  using etl::swap;
71  swap(_obj, other._obj);
72  swap(_callable, other._callable);
73  }
74 
75  /// Equivalent to return invoke(f, TETL_FORWARD(args)...);, where f is the
76  /// callable object referred to by *this, qualified with the same
77  /// cv-qualifiers as the function type Signature.
78  auto operator()(Args... args) const -> R { return _callable(_obj, TETL_FORWARD(args)...); }
79 };
