25 /// Converts value into a character string by successively filling the range
26 /// [first, last), where [first, last) is required to be a valid range.
27 ///
28 /// Integer formatters: value is converted to a string of digits in the given
29 /// base (with no redundant leading zeroes). Digits in the range 10..35
30 /// (inclusive) are represented as lowercase characters a..z. If value is less
31 /// than zero, the representation starts with a minus sign. The library provides
32 /// overloads for all signed and unsigned integer types and for the type char as
33 /// the type of the parameter value.
34 template <integral T>
35  requires(not same_as<T, bool>)
36 [[nodiscard]] constexpr auto to_chars(char* first, char* last, T val, int base = 10) -> to_chars_result
37 {
38  auto const len = static_cast<etl::size_t>(etl::distance(first, last));
39  auto const res = detail::integer_to_string<T>(val, first, base, len);
40  if (res.error == detail::integer_to_string_error::none) {
41  return to_chars_result{res.end, {}};
42  }
43  return to_chars_result{last, errc::value_too_large};
44 }
