14 /// \brief The enabled specializations of formatter define formatting rules for
15 /// a given type. Enabled specializations meet the Formatter requirements.
16 ///
17 /// https://en.cppreference.com/w/cpp/utility/format/formatter
18 template <typename T, typename CharT = char>
19 struct formatter {
20  formatter() = delete;
21  formatter(formatter const& other) = delete;
22  auto operator=(formatter const& other) -> formatter = delete;
23 };

25 /// \brief Standard specializations for basic type char.
26 template <>
27 struct formatter<char, char> {
28  template <typename FormatContext>
29  constexpr auto format(char val, FormatContext& fc) -> decltype(fc.out())
30  {
31  auto pos = fc.out();
32  *pos = val;
33  return pos++;
34  }
35 };

73 namespace detail {
74 template <typename Integer, typename FormatContext>
75 constexpr auto integer_format(Integer v, FormatContext& fc) -> decltype(fc.out())
76 {
77  char buf[32]{};
78  auto res = detail::integer_to_string(v, begin(buf), 10, sizeof(buf));
79  if (res.error == detail::integer_to_string_error::none) {
80  auto str = string_view{begin(buf)};
81  return formatter<string_view>().format(str, fc);
82  }
83  return formatter<string_view>().format("", fc);
84 }
85 } // namespace detail
