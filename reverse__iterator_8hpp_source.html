11 /// \brief reverse_iterator is an iterator adaptor that reverses the direction
12 /// of a given iterator. In other words, when provided with a bidirectional
13 /// iterator, `reverse_iterator` produces a new iterator that moves from the end
14 /// to the beginning of the sequence defined by the underlying bidirectional
15 /// iterator. This is the iterator returned by member functions `rbegin()` and
16 /// `rend()` of the standard library containers.
17 ///
18 /// https://en.cppreference.com/w/cpp/iterator/reverse_iterator
19 template <typename Iter>
20 struct reverse_iterator {
21  /// The underlying iterator type
22  using iterator_type = Iter;
23  /// The underlying value type
24  using value_type = typename iterator_traits<Iter>::value_type;
25  /// The type of subtracting to iterators
26  using difference_type = typename etl::iterator_traits<Iter>::difference_type;
27  /// The underlying reference type
28  using reference = typename etl::iterator_traits<Iter>::reference;
29  /// The underlying pointer type
30  using pointer = typename etl::iterator_traits<Iter>::pointer;
31 
32  /// \brief Constructs a new iterator adaptor.
33  ///
34  /// \details Default constructor. The underlying iterator is
35  /// value-initialized. Operations on the resulting iterator have defined
36  /// behavior if and only if the corresponding operations on a
37  /// value-initialized Iterator also have defined behavior.
38  constexpr reverse_iterator() : _current() { }
39 
40  /// \brief Constructs a new iterator adaptor.
41  ///
42  /// \details The underlying iterator is initialized with x.
43  constexpr explicit reverse_iterator(Iter x) : _current(x) { }
44 
45  /// \brief Constructs a new iterator adaptor.
46  ///
47  /// \details The underlying iterator is initialized with that of other.
48  template <typename Other>
49  constexpr reverse_iterator(reverse_iterator<Other> const& other) : _current(other.base())
50  {
51  }
52 
53  /// \brief The underlying iterator is assigned the value of the underlying
54  /// iterator of other, i.e. other.base().
55  template <typename Other>
56  constexpr auto operator=(reverse_iterator<Other> const& other) -> reverse_iterator&
57  {
58  _current = other.base();
59  return *this;
60  }

62  /// \brief Returns the underlying base iterator.
63  [[nodiscard]] constexpr auto base() const -> Iter { return _current; }
64 
65  /// \brief Returns a reference to the element previous to current.
66  constexpr auto operator*() const -> reference
67  {
68  auto tmp = _current;
69  return *--tmp;
70  }
71 
72  /// \brief Returns a pointer to the element previous to current.
73  constexpr auto operator->() const -> pointer { return etl::addressof(operator*()); }
74 
75  /// \brief Pre-increments by one respectively.
76  constexpr auto operator++() -> reverse_iterator&
77  {
78  --_current;

80  }

82  /// \brief Pre-increments by one respectively.
83  constexpr auto operator++(int) -> reverse_iterator
84  {
85  auto tmp(*this);
86  --_current;
87  return tmp;
88  }
89 
90  /// \brief Pre-decrements by one respectively.
91  constexpr auto operator--() -> reverse_iterator&
92  {
93  ++_current;
94  return *this;
95  }
96 
97  /// \brief Pre-decrements by one respectively.
98  constexpr auto operator--(int) -> reverse_iterator
99  {
100  auto tmp(*this);
101  ++_current;
102  return tmp;
103  }

105  /// \brief Returns an iterator which is advanced by n positions.
106  constexpr auto operator+(difference_type n) const -> reverse_iterator { return reverse_iterator(_current - n); }
107 
108  /// \brief Advances the iterator by n or -n positions respectively.
109  constexpr auto operator+=(difference_type n) -> reverse_iterator&
110  {
111  _current -= n;
112  return *this;
113  }
114 
115  /// \brief Returns an iterator which is advanced by -n positions.
116  constexpr auto operator-(difference_type n) const -> reverse_iterator { return reverse_iterator(_current + n); }
117 
118  /// \brief Advances the iterator by n or -n positions respectively.
119  constexpr auto operator-=(difference_type n) -> reverse_iterator&
120  {
121  _current += n;
122  return *this;
123  }
124 
125  /// \brief Returns a reference to the element at specified relative
126  /// location.
127  constexpr auto operator[](difference_type n) const -> reference { return *(*this + n); }
128 
129 private:
