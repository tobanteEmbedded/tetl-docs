11 /// \brief Class template time_point represents a point in time. It is
12 /// implemented as if it stores a value of type Duration indicating the time
13 /// interval from the start of the Clock's epoch.
14 ///
15 /// \tparam Clock Must meet the requirements for Clock
16 ///
17 /// https://en.cppreference.com/w/cpp/named_req/Clock
18 template <typename Clock, typename Duration = typename Clock::duration>
19 struct time_point {
20  /// \brief Clock, the clock on which this time point is measured.
21  using clock = Clock;
22 
23  /// \brief Duration, a duration type used to measure the time since epoch.
24  using duration = Duration;
25 
26  /// \brief Rep, an arithmetic type representing the number of ticks of the
27  /// duration.
28  using rep = typename duration::rep;
29 
30  /// \brief Period, a ratio type representing the tick period of the
31  /// duration.
32  using period = typename duration::period;
33 
34  /// \brief Constructs a new time_point from one of several optional data
35  /// sources. Default constructor, creates a time_point representing the
36  /// Clock's epoch (i.e., time_since_epoch() is zero).
37  constexpr time_point() noexcept = default;
38 
39  /// \brief Constructs a new time_point from one of several optional data
40  /// sources. Constructs a time_point at Clock's epoch plus d.
41  constexpr explicit time_point(duration const& d) noexcept : _d{d} { }
42 
43  /// \brief Constructs a new time_point from one of several optional data
44  /// sources. Constructs a time_point by converting t to duration. This
45  /// constructor only participates in overload resolution if Duration2 is
46  /// implicitly convertible to duration.
47  template <typename Dur2>
48  requires(is_convertible_v<Dur2, duration>)
49  constexpr time_point(time_point<clock, Dur2> const& t) : _d{t.time_since_epch()}
50  {
51  }
52 
53  /// \brief Returns a duration representing the amount of time between *this
54  /// and the clock's epoch.
55  [[nodiscard]] constexpr auto time_since_epoch() const noexcept -> duration { return _d; }
56 
57  /// \brief Modifies the time point by the given duration. Applies the offset
58  /// d to pt. Effectively, d is added to the internally stored duration d_ as
59  /// d_
60  /// += d.
61  constexpr auto operator+=(duration const& d) noexcept -> time_point&
62  {
63  _d += d;
64  return *this;
65  }
66 
67  /// \brief Modifies the time point by the given duration. Applies the offset
68  /// d to pt in negative direction. Effectively, d is subtracted from
69  /// internally stored duration d_ as d_ -= d.
70  constexpr auto operator-=(duration const& d) noexcept -> time_point&
71  {
72  _d -= d;
73  return *this;
74  }
75 
76  /// \brief Modifies the point in time *this represents by one tick of the
77  /// duration.
78  constexpr auto operator++() noexcept -> time_point&
79  {
80  ++_d;
81  return *this;
82  }
83 
84  /// \brief Modifies the point in time *this represents by one tick of the
85  /// duration.
86  constexpr auto operator++(int) noexcept -> time_point { return time_point(_d++); }
87 
88  /// \brief Modifies the point in time *this represents by one tick of the
89  /// duration.
90  constexpr auto operator--() noexcept -> time_point&
91  {
92  --_d;
93  return *this;
94  }
95 
96  /// \brief Modifies the point in time *this represents by one tick of the
97  /// duration.
98  constexpr auto operator--(int) noexcept -> time_point { return time_point(_d--); }
99 
100  /// \brief Returns a time_point with the smallest possible duration,
101  [[nodiscard]] static constexpr auto min() noexcept -> time_point { return time_point(duration::min()); }
102 
103  /// \brief Returns a time_point with the largest possible duration,
104  [[nodiscard]] static constexpr auto max() noexcept -> time_point { return time_point(duration::max()); }
105 
106 private:
107  duration _d{};
108 };
