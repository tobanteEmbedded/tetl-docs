40template <typename T>
41constexpr auto incomplete_beta_inv_initial_val_1_tval(T const p) noexcept -> T
42{ // a > 1.0
43 return (
44 p > T(0.5) ? // if
45 sqrt(-T(2) * log(T(1) - p))
46 :
47 // else
48 sqrt(-T(2) * log(p))
49 );
50}

52template <typename T>
53constexpr auto incomplete_beta_inv_initial_val_1_int_begin(T const tVal) noexcept -> T
54{ // internal for a > 1.0
55 return (
56 tVal
57 - (T(2.515517) + T(0.802853) * tVal + T(0.010328) * tVal * tVal)
58 / (T(1) + T(1.432788) * tVal + T(0.189269) * tVal * tVal + T(0.001308) * tVal * tVal * tVal)
59 );
60}

62template <typename T>
63constexpr auto incomplete_beta_inv_initial_val_1_int_ab1(T const alphaPar, T const betaPar) noexcept -> T
64{
65 return (T(1) / (2 * alphaPar - T(1)) + T(1) / (2 * betaPar - T(1)));
66}

68template <typename T>
69constexpr auto incomplete_beta_inv_initial_val_1_int_ab2(T const alphaPar, T const betaPar) noexcept -> T
70{
71 return (T(1) / (2 * betaPar - T(1)) - T(1) / (2 * alphaPar - T(1)));
72}

74template <typename T>
75constexpr auto incomplete_beta_inv_initial_val_1_int_h(T const abTerm1) noexcept -> T
76{
77 return (T(2) / abTerm1);

80template <typename T>
81constexpr auto incomplete_beta_inv_initial_val_1_int_w(T const value, T const abTerm2, T const hTerm) noexcept -> T
82{
83 // return( value * sqrt(h_term + lambda)/h_term - ab_term_2*(lambda
84 // + 5.0/6.0 -2.0/(3.0*h_term)) );
85 return (
86 value * sqrt(hTerm + (value * value - T(3)) / T(6)) / hTerm
87 - abTerm2 * ((value * value - T(3)) / T(6) + T(5) / T(6) - T(2) / (T(3) * hTerm))
88 );
89}

91template <typename T>
92constexpr auto incomplete_beta_inv_initial_val_1_int_end(T const alphaPar, T const betaPar, T const wTerm) noexcept -> T
93{
94 return (alphaPar / (alphaPar + betaPar * exp(2 * wTerm)));
95}

97template <typename T>
98constexpr auto
99incomplete_beta_inv_initial_val_1(T const alphaPar, T const betaPar, T const tVal, T const sgnTerm) noexcept -> T
100{ // a > 1.0
101 return incomplete_beta_inv_initial_val_1_int_end(
102 alphaPar,
103 betaPar,
104 incomplete_beta_inv_initial_val_1_int_w(
105 sgnTerm * incomplete_beta_inv_initial_val_1_int_begin(tVal),
106 incomplete_beta_inv_initial_val_1_int_ab2(alphaPar, betaPar),
107 incomplete_beta_inv_initial_val_1_int_h(incomplete_beta_inv_initial_val_1_int_ab1(alphaPar, betaPar))
108 )
109 );
110}

122constexpr auto incomplete_beta_inv_initial_val_2_s2(T const alphaPar, T const betaPar) noexcept -> T
123{
124 return (pow(betaPar / (alphaPar + betaPar), betaPar) / betaPar);

127template <typename T>
128constexpr auto
129incomplete_beta_inv_initial_val_2(T const alphaPar, T const betaPar, T const p, T const s1, T const s2) noexcept -> T
130{
131 return (
132 p <= s1 / (s1 + s2) ? pow(p * (s1 + s2) * alphaPar, T(1) / alphaPar)
133 : T(1) - pow(p * (s1 + s2) * betaPar, T(1) / betaPar)
134 );
135}

139template <typename T>
140constexpr auto incomplete_beta_inv_initial_val(T const alphaPar, T const betaPar, T const p) noexcept -> T
141{
142 return (
143 (alphaPar > T(1) && betaPar > T(1)) ?
144 // if
145 incomplete_beta_inv_initial_val_1(
146 alphaPar,
147 betaPar,
148 incomplete_beta_inv_initial_val_1_tval(p),
149 p < T(0.5) ? T(1) : T(-1)
150 )
151 :
152 // else
153 p > T(0.5) ?
154 // if
155 T(1)
156 - incomplete_beta_inv_initial_val_2(
157 betaPar,
158 alphaPar,
159 T(1) - p,
160 incomplete_beta_inv_initial_val_2_s1(betaPar, alphaPar),
161 incomplete_beta_inv_initial_val_2_s2(betaPar, alphaPar)
162 )
163 :
164 // else
165 incomplete_beta_inv_initial_val_2(
166 alphaPar,
167 betaPar,
168 p,
169 incomplete_beta_inv_initial_val_2_s1(alphaPar, betaPar),
170 incomplete_beta_inv_initial_val_2_s2(alphaPar, betaPar)
171 )
172 );
173}

184template <typename T>
185constexpr auto
186incomplete_beta_inv_deriv_1(T const value, T const alphaPar, T const betaPar, T const lbVal) noexcept -> T
187{ // derivative of the incomplete beta function w.r.t. x
188 return ( // indistinguishable from zero or one
189 etl::numeric_limits<T>::epsilon() > abs(value) ? T(0)
190 : etl::numeric_limits<T>::epsilon() > abs(T(1) - value) ? T(0)
191 :
192 // else
193 exp((alphaPar - T(1)) * log(value) + (betaPar - T(1)) * log(T(1) - value) - lbVal)
194 );
195}

197template <typename T>
198constexpr auto
199incomplete_beta_inv_deriv_2(T const value, T const alphaPar, T const betaPar, T const deriv1) noexcept -> T
200{ // second derivative of the incomplete beta function w.r.t. x
201 return (deriv1 * ((alphaPar - T(1)) / value - (betaPar - T(1)) / (T(1) - value)));
202}

219template <typename T>
220constexpr auto incomplete_beta_inv_halley(T const ratioVal1, T const ratioVal2) noexcept -> T
221{
222 return (ratioVal1 / max(T(0.8), min(T(1.2), T(1) - T(0.5) * ratioVal1 * ratioVal2)));
223}

225template <typename T>
226constexpr auto incomplete_beta_inv_recur(
227 T const value,
228 T const alphaPar,
229 T const betaPar,
230 T const p,
231 T const deriv1,
232 T const lbVal,
233 int const iterCount
234) noexcept -> T
235{
236 return ( // derivative = 0
237 etl::numeric_limits<T>::epsilon() > abs(deriv1)
238 ? incomplete_beta_inv_decision(value, alphaPar, betaPar, p, T(0), lbVal, GCEM_INCML_BETA_INV_MAX_ITER + 1)
239 :
240 // else
241 incomplete_beta_inv_decision(
242 value,
243 alphaPar,
244 betaPar,
245 p,
246 incomplete_beta_inv_halley(
247 incomplete_beta_inv_ratio_val_1(value, alphaPar, betaPar, p, deriv1),
248 incomplete_beta_inv_ratio_val_2(value, alphaPar, betaPar, deriv1)
249 ),
250 lbVal,
251 iterCount
252 )
253 );
254}

256template <typename T>
257constexpr auto incomplete_beta_inv_decision(
258 T const value,
259 T const alphaPar,
260 T const betaPar,
261 T const p,
262 T const direc,
263 T const lbVal,
264 int const iterCount
265) noexcept -> T
266{
267 return (
268 iterCount <= GCEM_INCML_BETA_INV_MAX_ITER ?

270 incomplete_beta_inv_recur(

275 incomplete_beta_inv_deriv_1(value, alphaPar, betaPar, lbVal),
276 lbVal,
277 iterCount + 1
278 )
279 :
280 // else
281 value - direc
282 );

300template <typename T>
301constexpr auto incomplete_beta_inv_check(T const alphaPar, T const betaPar, T const p) noexcept -> T
302{
303 return ( // NaN check
304 any_nan(alphaPar, betaPar, p) ? etl::numeric_limits<T>::quiet_NaN() :
305 // indistinguishable from zero or one
306 etl::numeric_limits<T>::epsilon() > p ? T(0)
307 : etl::numeric_limits<T>::epsilon() > abs(T(1) - p) ? T(1)
308 :
309 // else
310 incomplete_beta_inv_begin(
311 incomplete_beta_inv_initial_val(alphaPar, betaPar, p),
312 alphaPar,
313 betaPar,
314 p,
315 lbeta(alphaPar, betaPar)
316 )
317 );
318}
