32 template <typename T>
33 constexpr auto incomplete_gamma_quad_inp_vals(const T lb, const T ub, const int counter) noexcept -> T
34 {
35  return (ub - lb) * gauss_legendre_50_points[counter] / T(2) + (ub + lb) / T(2);
36 }

38 template <typename T>
39 constexpr auto incomplete_gamma_quad_weight_vals(const T lb, const T ub, const int counter) noexcept -> T
40 {
41  return (ub - lb) * gauss_legendre_50_weights[counter] / T(2);
42 }

44 template <typename T>
45 constexpr auto incomplete_gamma_quad_fn(const T x, const T a, const T lgTerm) noexcept -> T
46 {
47  return exp(-x + (a - T(1)) * log(x) - lgTerm);
48 }

50 template <typename T>
51 constexpr auto incomplete_gamma_quad_recur(
52  const T lb, const T ub, const T a, const T lgTerm, const int counter) noexcept -> T
53 {
54  return (counter < 49 ? // if
55  incomplete_gamma_quad_fn(incomplete_gamma_quad_inp_vals(lb, ub, counter), a, lgTerm)
56  * incomplete_gamma_quad_weight_vals(lb, ub, counter)
57  + incomplete_gamma_quad_recur(lb, ub, a, lgTerm, counter + 1)
58  :
59  // else
60  incomplete_gamma_quad_fn(incomplete_gamma_quad_inp_vals(lb, ub, counter), a, lgTerm)
61  * incomplete_gamma_quad_weight_vals(lb, ub, counter));
62 }

64 template <typename T>
65 constexpr auto incomplete_gamma_quad_lb(const T a, const T z) noexcept -> T
66 {
67  return (a > T(1000) ? max(T(0), min(z, a) - 11 * sqrt(a)) : // break integration into ranges
68  a > T(800) ? max(T(0), min(z, a) - 11 * sqrt(a))
69  : a > T(500) ? max(T(0), min(z, a) - 10 * sqrt(a))
70  : a > T(300) ? max(T(0), min(z, a) - 10 * sqrt(a))
71  : a > T(100) ? max(T(0), min(z, a) - 9 * sqrt(a))
72  : a > T(90) ? max(T(0), min(z, a) - 9 * sqrt(a))
73  : a > T(70) ? max(T(0), min(z, a) - 8 * sqrt(a))
74  : a > T(50) ? max(T(0), min(z, a) - 7 * sqrt(a))

107 template <typename T>
108 constexpr auto incomplete_gamma_cf_2_recur(const T a, const T z, const int depth) noexcept -> T
109 {
110  return (depth < 100 ? // if
111  (1 + (depth - 1) * 2 - a + z) + depth * (a - depth) / incomplete_gamma_cf_2_recur(a, z, depth + 1)
112  :
113  // else
114  (1 + (depth - 1) * 2 - a + z));
115 }

117 template <typename T>
118 constexpr auto incomplete_gamma_cf_2(const T a, const T z) noexcept -> T
119 { // lower (regularized) incomplete gamma function
120  return (T(1.0) - exp(a * log(z) - z - lgamma(a)) / incomplete_gamma_cf_2_recur(a, z, 1));

126 template <typename T>
127 constexpr auto incomplete_gamma_cf_1_coef(const T a, const T z, const int depth) noexcept -> T
128 {
129  return (is_odd(depth) ? -(a - 1 + T(depth + 1) / T(2)) * z : T(depth) / T(2) * z);
130 }

132 template <typename T>
133 constexpr auto incomplete_gamma_cf_1_recur(const T a, const T z, const int depth) noexcept -> T
134 {
135  return (depth < GCEM_INCML_GAMMA_MAX_ITER ? // if
136  (a + depth - 1) + incomplete_gamma_cf_1_coef(a, z, depth) / incomplete_gamma_cf_1_recur(a, z, depth + 1)
137  :
138  // else
139  (a + depth - 1));
140 }

143 constexpr auto incomplete_gamma_cf_1(const T a, const T z) noexcept -> T
144 { // lower (regularized) incomplete gamma function
145  return (exp(a * log(z) - z - lgamma(a)) / incomplete_gamma_cf_1_recur(a, z, 1));

150 template <typename T>
151 constexpr auto incomplete_gamma_check(const T a, const T z) noexcept -> T
152 {
153  return ( // NaN check
154  any_nan(a, z) ? etl::numeric_limits<T>::quiet_NaN() :
155  //
156  a < T(0) ? etl::numeric_limits<T>::quiet_NaN()
157  :
158  //
159  etl::numeric_limits<T>::epsilon() > z ? T(0)
160  :
161  //
162  etl::numeric_limits<T>::epsilon() > a ? T(1)
163  :
164  // cf or quadrature
165  (a < T(10)) && (z - a < T(10)) ? incomplete_gamma_cf_1(a, z)
166  : (a < T(10)) || (z / a > T(3)) ? incomplete_gamma_cf_2(a, z)
167  :
168  // else
169  incomplete_gamma_quad(a, z));
170 }
