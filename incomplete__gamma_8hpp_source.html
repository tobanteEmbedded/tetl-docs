30// 50 point Gauss-Legendre quadrature
31
32template <typename T>
33constexpr auto incomplete_gamma_quad_inp_vals(T const lb, T const ub, int const counter) noexcept -> T
34{
35 return (ub - lb) * gauss_legendre_50_points[counter] / T(2) + (ub + lb) / T(2);
36}
37
38template <typename T>
39constexpr auto incomplete_gamma_quad_weight_vals(T const lb, T const ub, int const counter) noexcept -> T
40{
41 return (ub - lb) * gauss_legendre_50_weights[counter] / T(2);
42}
43
44template <typename T>
45constexpr auto incomplete_gamma_quad_fn(T const x, T const a, T const lgTerm) noexcept -> T
46{
47 return exp(-x + (a - T(1)) * log(x) - lgTerm);
48}
49
50template <typename T>
51constexpr auto
52incomplete_gamma_quad_recur(T const lb, T const ub, T const a, T const lgTerm, int const counter) noexcept -> T
53{
54 return (
55 counter < 49 ? // if
56 incomplete_gamma_quad_fn(incomplete_gamma_quad_inp_vals(lb, ub, counter), a, lgTerm)
57 * incomplete_gamma_quad_weight_vals(lb, ub, counter)
58 + incomplete_gamma_quad_recur(lb, ub, a, lgTerm, counter + 1)
59 :
60 // else
61 incomplete_gamma_quad_fn(incomplete_gamma_quad_inp_vals(lb, ub, counter), a, lgTerm)
62 * incomplete_gamma_quad_weight_vals(lb, ub, counter)
63 );
64}

66template <typename T>
67constexpr auto incomplete_gamma_quad_lb(T const a, T const z) noexcept -> T
68{
69 return (
70 a > T(1000) ? max(T(0), min(z, a) - 11 * sqrt(a)) : // break integration into ranges
71 a > T(800) ? max(T(0), min(z, a) - 11 * sqrt(a))
72 : a > T(500) ? max(T(0), min(z, a) - 10 * sqrt(a))
73 : a > T(300) ? max(T(0), min(z, a) - 10 * sqrt(a))
74 : a > T(100) ? max(T(0), min(z, a) - 9 * sqrt(a))
75 : a > T(90) ? max(T(0), min(z, a) - 9 * sqrt(a))
76 : a > T(70) ? max(T(0), min(z, a) - 8 * sqrt(a))
77 : a > T(50) ? max(T(0), min(z, a) - 7 * sqrt(a))
78 : a > T(40) ? max(T(0), min(z, a) - 6 * sqrt(a))
79 : a > T(30) ? max(T(0), min(z, a) - 5 * sqrt(a))
80 :
81 // else
82 max(T(0), min(z, a) - 4 * sqrt(a))
83 );
84}

86template <typename T>
87constexpr auto incomplete_gamma_quad_ub(T const a, T const z) noexcept -> T
88{
89 return (
90 a > T(1000) ? min(z, a + 10 * sqrt(a))
91 : a > T(800) ? min(z, a + 10 * sqrt(a))
92 : a > T(500) ? min(z, a + 9 * sqrt(a))
93 : a > T(300) ? min(z, a + 9 * sqrt(a))
94 : a > T(100) ? min(z, a + 8 * sqrt(a))
95 : a > T(90) ? min(z, a + 8 * sqrt(a))
96 : a > T(70) ? min(z, a + 7 * sqrt(a))
97 : a > T(50) ? min(z, a + 6 * sqrt(a))
98 :
99 // else
100 min(z, a + 5 * sqrt(a))
101 );
102}

104template <typename T>
105constexpr auto incomplete_gamma_quad(T const a, T const z) noexcept -> T
106{
107 return incomplete_gamma_quad_recur(incomplete_gamma_quad_lb(a, z), incomplete_gamma_quad_ub(a, z), a, lgamma(a), 0);
108}

110// reverse cf expansion
111// see: https://functions.wolfram.com/GammaBetaErf/Gamma2/10/0003/
112
113template <typename T>
114constexpr auto incomplete_gamma_cf_2_recur(T const a, T const z, int const depth) noexcept -> T
115{
116 return (
117 depth < 100 ? // if
118 (1 + (depth - 1) * 2 - a + z) + depth * (a - depth) / incomplete_gamma_cf_2_recur(a, z, depth + 1)
119 :
120 // else
121 (1 + (depth - 1) * 2 - a + z)
122 );
123}
124
125template <typename T>
126constexpr auto incomplete_gamma_cf_2(T const a, T const z) noexcept -> T
127{ // lower (regularized) incomplete gamma function
128 return (T(1.0) - exp(a * log(z) - z - lgamma(a)) / incomplete_gamma_cf_2_recur(a, z, 1));
129}

131// cf expansion
132// see: http://functions.wolfram.com/GammaBetaErf/Gamma2/10/0009/
133
134template <typename T>
135constexpr auto incomplete_gamma_cf_1_coef(T const a, T const z, int const depth) noexcept -> T
136{
137 return (is_odd(depth) ? -(a - 1 + T(depth + 1) / T(2)) * z : T(depth) / T(2) * z);
138}
139
140template <typename T>
141constexpr auto incomplete_gamma_cf_1_recur(T const a, T const z, int const depth) noexcept -> T
142{
143 return (
144 depth < GCEM_INCML_GAMMA_MAX_ITER ? // if
145 (a + depth - 1) + incomplete_gamma_cf_1_coef(a, z, depth) / incomplete_gamma_cf_1_recur(a, z, depth + 1)
146 :
147 // else
148 (a + depth - 1)
149 );
150}
151
152template <typename T>
153constexpr auto incomplete_gamma_cf_1(T const a, T const z) noexcept -> T
154{ // lower (regularized) incomplete gamma function
155 return (exp(a * log(z) - z - lgamma(a)) / incomplete_gamma_cf_1_recur(a, z, 1));
156}

160template <typename T>
161constexpr auto incomplete_gamma_check(T const a, T const z) noexcept -> T
162{
163 return ( // NaN check
164 any_nan(a, z) ? etl::numeric_limits<T>::quiet_NaN() :
165 //
166 a < T(0) ? etl::numeric_limits<T>::quiet_NaN()
167 :
168 //
169 etl::numeric_limits<T>::epsilon() > z ? T(0)
170 :
171 //
172 etl::numeric_limits<T>::epsilon() > a ? T(1)
173 :
174 // cf or quadrature
175 (a < T(10)) && (z - a < T(10)) ? incomplete_gamma_cf_1(a, z)
176 : (a < T(10)) || (z / a > T(3)) ? incomplete_gamma_cf_2(a, z)
177 :
178 // else
179 incomplete_gamma_quad(a, z)
180 );
181}
