17 /// \brief Provide pointer_like_traits for non-cvr pointers.
18 template <typename T>
19 struct pointer_like_traits<T*> {
20  [[nodiscard]] static auto get_as_void_pointer(T* p) -> void* { return p; }
21 
22  [[nodiscard]] static auto get_from_void_pointer(void* p) -> T* { return static_cast<T*>(p); }
23 
24  static constexpr size_t free_bits = detail::log2(alignof(T));
25 };
