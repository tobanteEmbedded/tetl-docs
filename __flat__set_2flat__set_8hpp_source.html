28 /// \brief A flat_set is a container adaptor that provides an associative
29 /// container interface that supports unique keys (contains at most one of each
30 /// key value) and provides for fast retrieval of the keys themselves. flat_set
31 /// supports random access iterators. Any sequence container supporting random
32 /// access iteration can be used to instantiate flat_set
33 ///
34 /// \details A flat_set satisfies all of the requirements of a container and of
35 /// a reversible container. flat_set satisfies the requirements of an
36 /// associative container, except that:
37 ///
38 /// - it does not meet the requirements related to node handles,
39 /// - it does not meet the requirements related to iterator invalidation, and
40 /// - the time complexity of the insert, emplace, emplace_hint, and erase
41 /// members that respectively insert, emplace or erase a single element from the
42 /// set is linear, including the ones that take an insertion position iterator.
43 ///
44 /// https://isocpp.org/files/papers/P1222R1.pdf
45 ///
46 /// https://youtu.be/b9ZYM0d6htg

144  // 21.6.5.3, modifiers
145  template <typename... Args>
146  constexpr auto emplace(Args&&... args) -> etl::pair<iterator, bool>
147  {
148  auto key = Key{TETL_FORWARD(args)...};
149  iterator it = lower_bound(key);
150 
151  if (it == end() || _compare(key, *it)) {
152  it = _container.emplace(it, TETL_MOVE(key));
153  return etl::make_pair(it, true);
154  }
155 
156  return etl::make_pair(it, false);
157  }
