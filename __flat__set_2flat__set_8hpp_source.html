165 {
166 auto key = Key{etl::forward<Args>(args)...};

405constexpr auto erase_if(etl::flat_set<Key, Container, Compare>& c, Pred pred) ->
406 typename etl::flat_set<Key, Container, Compare>::size_type
407{
