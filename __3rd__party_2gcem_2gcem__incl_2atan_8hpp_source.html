36 template <typename T>
37 constexpr auto atan_series_order_calc(T const x, T const xPow, uint_t const order) noexcept -> T
38 {
39  return (T(1) / (T((order - 1) * 4 - 1) * xPow) - T(1) / (T((order - 1) * 4 + 1) * xPow * x));
40 }

42 template <typename T>
43 constexpr auto atan_series_order(T const x, T const xPow, uint_t const order, uint_t const maxOrder) noexcept -> T
44 {
45  return static_cast<T>(
46  order == 1 ? GCEM_HALF_PI - T(1) / x + atan_series_order(x * x, pow(x, 3), order + 1, maxOrder) :
47  // NOTE: x changes to x*x for order > 1
48  order < maxOrder
49  ? atan_series_order_calc(x, xPow, order) + atan_series_order(x, xPow * x * x, order + 1, maxOrder)
50  :
51  // order == max_order
52  atan_series_order_calc(x, xPow, order)
53  );
54 }

56 template <typename T>
57 constexpr auto atan_series_main(T const x) noexcept -> T
58 {
59  return static_cast<T>(
60  x < T(3) ? atan_series_order(x, x, 1U, 10U) : // O(1/x^39)
61  x < T(4) ? atan_series_order(x, x, 1U, 9U)
62  : // O(1/x^35)
63  x < T(5) ? atan_series_order(x, x, 1U, 8U)
64  : // O(1/x^31)
65  x < T(7) ? atan_series_order(x, x, 1U, 7U)
66  : // O(1/x^27)
67  x < T(11) ? atan_series_order(x, x, 1U, 6U)
68  : // O(1/x^23)
69  x < T(25) ? atan_series_order(x, x, 1U, 5U)
70  : // O(1/x^19)
71  x < T(100) ? atan_series_order(x, x, 1U, 4U)
72  : // O(1/x^15)
73  x < T(1000) ? atan_series_order(x, x, 1U, 3U)
74  : // O(1/x^11)
75  atan_series_order(x, x, 1U, 2U)
76  ); // O(1/x^7)

81 template <typename T>
82 constexpr auto atan_cf_recur(T const xx, uint_t const depth, uint_t const maxDepth) noexcept -> T
83 {
84  return (
85  depth < maxDepth ? // if
86  T(2 * depth - 1) + depth * depth * xx / atan_cf_recur(xx, depth + 1, maxDepth)
87  :
88  // else
89  T(2 * depth - 1)
90  );
91 }

93 template <typename T>
94 constexpr auto atan_cf_main(T const x) noexcept -> T
95 {
96  return (
97  x < T(0.5) ? x / atan_cf_recur(x * x, 1U, 15U)
98  : x < T(1) ? x / atan_cf_recur(x * x, 1U, 25U)
99  : x < T(1.5) ? x / atan_cf_recur(x * x, 1U, 35U)
100  : x < T(2) ? x / atan_cf_recur(x * x, 1U, 45U)
101  : x / atan_cf_recur(x * x, 1U, 52U)
102  );
103 }

113 template <typename T>
114 constexpr auto atan_check(T const x) noexcept -> T
115 {
116  return ( // NaN check
117  is_nan(x) ? etl::numeric_limits<T>::quiet_NaN() :
118  // indistinguishable from zero
119  etl::numeric_limits<T>::epsilon() > abs(x) ? T(0)
120  :
121  // negative or positive
122  x < T(0) ? -atan_begin(-x)
123  : atan_begin(x)
124  );
