17 /// \brief The class template bitset represents a fixed-size sequence of N bits.
18 /// Bitsets can be manipulated by standard logic operators.
19 /// \bug Add tests for sizes that are not a power of two. Broken at the moment.

21 template <etl::size_t N>
22 struct bitset {
23  /// \brief The primary use of etl::bitset::reference is to provide an
24  /// l-value that can be returned from operator[].
25  ///
26  /// \details This class is used as a proxy object to allow users to interact
27  /// with individual bits of a bitset, since standard C++ types (like
28  /// references and pointers) are not built with enough precision to specify
29  /// individual bits.
30  struct reference {
31  /// \brief Assigns a value to the referenced bit.
32  constexpr auto operator=(bool value) noexcept -> reference&
33  {
34  if (value) {
35  *_data |= (1U << _position);
36  return *this;
37  }
38 
39  *_data &= ~(1U << _position);
40  return *this;
41  }
42 
43  /// \brief Assigns a value to the referenced bit.
44  /// \returns *this
45  constexpr auto operator=(reference const& x) noexcept -> reference&
46  {
47  (*this) = static_cast<bool>(x);
48  return *this;
49  }
50 
51  /// \brief Returns the value of the referenced bit.
52  [[nodiscard]] constexpr operator bool() const noexcept { return (*_data & (1U << _position)) != 0; }
53 
54  /// \brief Returns the inverse of the referenced bit.
55  [[nodiscard]] constexpr auto operator~() const noexcept -> bool { return !static_cast<bool>(*this); }
56 
57  /// \brief Inverts the referenced bit.
58  /// \returns *this
59  constexpr auto flip() noexcept -> reference&
60  {
61  *_data ^= 1U << _position;
62  return *this;
63  }

65  private:
66  constexpr explicit reference(uint8_t* data, uint8_t position) : _data{data}, _position{position} { }

76  /// \brief Constructs a bitset, initializing the first (rightmost, least
77  /// significant) M bit positions to the corresponding bit values of val,
78  /// where M is the smaller of the number of bits in an unsigned long long
79  /// and the number of bits N in the bitset being constructed. If M is less
80  /// than N (the bitset is longer than 64 bits, for typical implementations
81  /// of unsigned long long), the remaining bit positions are initialized to
82  /// zeroes.
83  constexpr bitset(unsigned long long val) noexcept
84  {
85  auto const n = min<size_t>(numeric_limits<decltype(val)>::digits, size());
86  for (size_t i = 0; i < n; ++i) {
87  if (((val >> i) & 1U) == 1U) {
88  set(i);
89  }
90  }
91  }

93  /// \brief Constructs a bitset using the characters in the
94  /// etl::basic_string_view str.
95  ///
96  /// \details An optional starting position pos and length n can be provided,
97  /// as well as characters denoting alternate values for set (one) and unset
98  /// (zero) bits. Traits::eq() is used to compare the character values. The
99  /// effective length of the initializing string is min(n, str.size() - pos).
100  ///
101  /// \param str string used to initialize the bitset
102  /// \param pos a starting offset into str
103  /// \param n number of characters to use from str
104  /// \param zero alternate character for set bits in str
105  /// \param one alternate character for unset bits in str
106  template <typename CharT, typename Traits>
107  explicit constexpr bitset(
108  basic_string_view<CharT, Traits> const& str,
109  typename basic_string_view<CharT, Traits>::size_type pos = 0,
110  typename basic_string_view<CharT, Traits>::size_type n = basic_string_view<CharT, Traits>::npos,
111  CharT zero = CharT('0'),
112  CharT one = CharT('1')
113  )
114  : bitset(0ULL)
115  {
116  auto const len = min<decltype(pos)>(n, str.size() - pos);
117  TETL_ASSERT(len >= 0);
118  TETL_ASSERT(len <= size());
119 
120  for (decltype(pos) i = 0; i < len; ++i) {
121  if (Traits::eq(str[i + pos], one)) {
122  set(i, true);
123  }
124  if (Traits::eq(str[i + pos], zero)) {
125  set(i, false);
126  }
127  }
128  }

319  /// \brief Converts the contents of the bitset to a string. Uses zero to
320  /// represent bits with value of false and one to represent bits with value
321  /// of true. The resulting string contains N characters with the first
322  /// character corresponds to the last (N-1th) bit and the last character
323  /// corresponding to the first bit.
324  template <size_t Capacity, typename CharT = char, typename Traits = char_traits<CharT>>
325  [[nodiscard]] constexpr auto
326  to_string(CharT zero = CharT('0'), CharT one = CharT('1')) const -> basic_static_string<CharT, Capacity, Traits>
327  {
328  // TODO: [tobi] This currently truncates the low bits, if the string is
329  // large enough.
330  auto str = basic_static_string<CharT, Capacity, Traits>{};
331  for (auto i{size() - 1U}; i != 0; --i) {
332  str.push_back(test(i) ? one : zero);
333  }
334  str.push_back(test(0) ? one : zero);
335  return str;
336  }

373  auto const idx = etl::min<UInt>(static_cast<UInt>(size()), digits);
