108 template <typename CharT, typename Traits>
109 explicit constexpr bitset(
110 basic_string_view<CharT, Traits> const& str,
111 typename basic_string_view<CharT, Traits>::size_type pos = 0,
112 typename basic_string_view<CharT, Traits>::size_type n = basic_string_view<CharT, Traits>::npos,
113 CharT zero = CharT('0'),
114 CharT one = CharT('1')
115 )
116 : bitset(0ULL)
117 {
118 auto const len = min<decltype(pos)>(n, str.size() - pos);
119 TETL_ASSERT(len >= 0);
120 TETL_ASSERT(len <= size());
121
122 for (decltype(pos) i = 0; i < len; ++i) {
123 if (Traits::eq(str[i + pos], one)) {
124 set(i, true);
125 }
126 if (Traits::eq(str[i + pos], zero)) {
127 set(i, false);
128 }
129 }
130 }

328 to_string(CharT zero = CharT('0'), CharT one = CharT('1')) const -> basic_static_string<CharT, Capacity, Traits>
329 {
