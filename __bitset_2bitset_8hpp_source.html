55 template <typename CharT, typename Traits>
56 explicit constexpr bitset(
57 basic_string_view<CharT, Traits> const& str,
58 typename basic_string_view<CharT, Traits>::size_type pos = 0,
59 typename basic_string_view<CharT, Traits>::size_type n = basic_string_view<CharT, Traits>::npos,
60 CharT zero = CharT('0'),
61 CharT one = CharT('1')
62 )
63 : bitset(0ULL)
64 {
65 auto const len = etl::min<decltype(pos)>(n, str.size() - pos);
66 TETL_PRECONDITION(len >= 0);
67 TETL_PRECONDITION(len <= size());
68
69 for (decltype(pos) i = 0; i < len; ++i) {
70 if (Traits::eq(str[i + pos], one)) {
71 set(i, true);
72 }
73 if (Traits::eq(str[i + pos], zero)) {
74 set(i, false);
75 }
76 }
77 }

238 template <size_t Capacity, typename CharT = char, typename Traits = char_traits<CharT>>
239 requires(Capacity >= Bits)
240 [[nodiscard]] constexpr auto
241 to_string(CharT zero = CharT('0'), CharT one = CharT('1')) const -> basic_inplace_string<CharT, Capacity, Traits>
242 {
243 auto str = basic_inplace_string<CharT, Capacity, Traits>{};
244 for (auto i{size() - 1U}; i != 0; --i) {
245 str.push_back(test(i) ? one : zero);
246 }
247 str.push_back(test(0) ? one : zero);
248 return str;
249 }

288 template <typename UInt>
289 [[nodiscard]] constexpr auto to_unsigned_type() const noexcept -> UInt
290 {
291 constexpr auto digits = static_cast<UInt>(etl::numeric_limits<UInt>::digits);
292 auto const idx = etl::min<UInt>(static_cast<UInt>(size()), digits);
293 UInt result{};
294 for (UInt i{0}; i != idx; ++i) {
295 if (test(static_cast<etl::size_t>(i))) {
296 result = etl::set_bit(result, i);
297 }
298 }
299 return result;
