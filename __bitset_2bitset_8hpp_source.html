112 template <typename CharT, typename Traits>
113 explicit constexpr bitset(
114 basic_string_view<CharT, Traits> const& str,
115 typename basic_string_view<CharT, Traits>::size_type pos = 0,
116 typename basic_string_view<CharT, Traits>::size_type n = basic_string_view<CharT, Traits>::npos,
117 CharT zero = CharT('0'),
118 CharT one = CharT('1')
119 )
120 : bitset(0ULL)
121 {
122 auto const len = min<decltype(pos)>(n, str.size() - pos);
123 TETL_ASSERT(len >= 0);
124 TETL_ASSERT(len <= size());
125
126 for (decltype(pos) i = 0; i < len; ++i) {
127 if (Traits::eq(str[i + pos], one)) {
128 set(i, true);
129 }
130 if (Traits::eq(str[i + pos], zero)) {
131 set(i, false);
132 }
133 }
134 }

332 to_string(CharT zero = CharT('0'), CharT one = CharT('1')) const -> basic_inplace_string<CharT, Capacity, Traits>
333 {
334 auto str = basic_inplace_string<CharT, Capacity, Traits>{};
