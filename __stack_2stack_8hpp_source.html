19 /// \brief The stack class is a container adapter that gives the programmer
20 /// the functionality of a stack - specifically, a LIFO (last-in, first-out)
21 /// data structure.
22 ///
23 /// \details The class template acts as a wrapper to the underlying container -
24 /// only a specific set of functions is provided. The stack pushes and pops the
25 /// element from the back of the underlying container, known as the top of the
26 /// stack.
27 template <typename T, typename Container>
28 struct stack {
29  using value_type = typename Container::value_type;
30  using reference = typename Container::reference;
31  using const_reference = typename Container::const_reference;
32  using size_type = typename Container::size_type;
33  using container_type = Container;
34

36  constexpr stack() : stack{Container{}} { }

39  /// cont.
40  constexpr explicit stack(Container const& cont) : c{cont} { }
41

43  constexpr explicit stack(Container&& cont) : c{TETL_MOVE(cont)} { }

88  /// \brief Pushes a new element on top of the stack. The element is
89  /// constructed in-place, i.e. no copy or move operations are performed. The
