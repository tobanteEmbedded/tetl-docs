327 template <typename Other>
328 constexpr variant(Other&& other, [[maybe_unused]] copy_move_tag tag)
329 : _index()
330 , _union(uninitialized_union())
331 {
332 etl::visit_with_index([&](auto param) {
333 replace(param.index, etl::move(param).value());
334 }, etl::forward<Other>(other));
335 }
336
337 template <typename Other>
338 constexpr auto assign(Other&& other) -> void
339 {
340 etl::visit_with_index([&](auto lhs, auto rhs) {
341 if constexpr (lhs.index == rhs.index) {
342 lhs.value() = etl::move(rhs.value());
343 } else {
344 destroy();
345 replace(rhs.index, etl::move(rhs.value()));
346 }
347 }, *this, etl::forward<Other>(other));
348 }
349
350 template <typename... Args>
351 constexpr auto replace(auto index, Args&&... args) -> auto&
352 {
353 etl::construct_at(etl::addressof(_union), index, etl::forward<Args>(args)...);
354 _index = static_cast<index_type>(index.value);
355 return (*this)[index];
356 }
357
358 constexpr auto destroy() -> void
359 {
360 etl::visit([](auto& v) { etl::destroy_at(etl::addressof(v)); }, *this);
361 }
