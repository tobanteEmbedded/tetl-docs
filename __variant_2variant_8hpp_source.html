350 constexpr auto swap(variant& rhs
351 ) noexcept(((is_nothrow_move_constructible_v<Ts> && is_nothrow_swappable_v<Ts>) && ...)) -> void
352 {
353 if (index() == rhs.index()) {
354 return etl::visit([](auto& l, auto& r) -> void {
355 if constexpr (etl::is_same_v<decltype(l), decltype(r)>) {
356 using etl::swap;
357 swap(l, r);
358 } else {
359 etl::unreachable();
360 }
361 }, *this, rhs);
362 }
363
364 etl::swap(_data, rhs._data);
365 etl::swap(_index, rhs._index);
366 }
