348 template <typename Other>
349 constexpr variant(Other&& other, [[maybe_unused]] copy_move_tag tag)
350 : _index()
351 , _union(etl::uninitialized_union())
352 {
353 etl::visit_with_index([&](auto param) {
354 replace(param.index, etl::move(param).value());
355 }, etl::forward<Other>(other));
356 }
357
358 template <typename Other>
359 constexpr auto assign(Other&& other) -> void
360 {
361 etl::visit_with_index([&](auto lhs, auto rhs) {
362 if constexpr (lhs.index == rhs.index) {
363 lhs.value() = etl::move(rhs.value());
364 } else {
365 destroy();
366 replace(rhs.index, etl::move(rhs.value()));
367 }
368 }, *this, etl::forward<Other>(other));
369 }

371 template <typename... Args>
372 constexpr auto replace(auto index, Args&&... args) -> auto&
373 {
374 etl::construct_at(etl::addressof(_union), index, etl::forward<Args>(args)...);
375 _index = static_cast<index_type>(index.value);
376 return (*this)[index];
377 }

379 constexpr auto destroy() -> void
380 {
381 etl::visit([](auto& v) { etl::destroy_at(etl::addressof(v)); }, *this);
382 }
